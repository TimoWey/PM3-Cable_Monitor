47/** ***************************************************************************
48 * @brief Main function
49 * @return not used because main ends in an infinite loop

51 * Initialization and infinite while loop

53int main(void) {
54 HAL_Init(); // Initialize the system
55
56 SystemClock_Config(); // Configure system clocks
57
58 GPIO_Init(); // Init PE5 and set it to HIGH
59
60#ifdef FLIPPED_LCD
61 BSP_LCD_Init_Flipped(); // Initialize the LCD for flipped orientation
62#else
63 BSP_LCD_Init(); // Initialize the LCD display
64#endif
65 BSP_LCD_LayerDefaultInit(LCD_FOREGROUND_LAYER, LCD_FRAME_BUFFER);
66 BSP_LCD_SelectLayer(LCD_FOREGROUND_LAYER);
67 BSP_LCD_DisplayOn();
68 BSP_LCD_Clear(LCD_COLOR_WHITE);
69
70 BSP_TS_Init(BSP_LCD_GetXSize(), BSP_LCD_GetYSize()); // Touchscreen
71 /* Uncomment next line to enable touchscreen interrupt */
72 // BSP_TS_ITConfig(); // Enable Touchscreen interrupt
73
74 PB_init(); // Initialize the user pushbutton
75 PB_enableIRQ(); // Enable interrupt on user pushbutton
76
77 BSP_LED_Init(LED3); // Toggles in while loop
78 BSP_LED_Init(LED4); // Is toggled by user button
79
80 MENU_draw(); // Draw the menu
81 MENU_hint(); // Show hint at startup
82
83 gyro_disable(); // Disable gyro, use those analog inputs
84
85 MEAS_GPIO_analog_init(); // Configure GPIOs in analog mode
86 MEAS_timer_init(); // Configure the timer
87
88
89
90 /* Infinite while loop */
91 while (1) { // Infinitely loop in main function
92 BSP_LED_Toggle(LED3); // Visual feedback when running
93
94 if (MEAS_data_ready) { // Show data if new data available
95 MEAS_data_ready = false;
96 MEAS_show_data();
97 }
98
99 if (PB_pressed()) { // Check if user pushbutton was pressed
100 DAC_active = !DAC_active; // Toggle DAC on/off
101 if (DAC_active) {
102 DAC_init();
103 BSP_LED_On(LED4);
104 } else {
105 DAC_reset();
106 BSP_LED_Off(LED4);
107 }
108 }
109
110 // Check if Device should power off (PE3)
111 if (HAL_GPIO_ReadPin(GPIOE, PIN_PE3) == GPIO_PIN_SET)
112 {
113 // Set PE5 low
114 HAL_GPIO_WritePin(GPIOE, PIN_PE5, GPIO_PIN_RESET);
115
116 // Delay for 3 seconds
117 HAL_Delay(3000);
118 }
119
120 /* Comment next line if touchscreen interrupt is enabled */
121 MENU_check_transition();
122
123 switch (MENU_get_transition()) { // Handle user menu choice
124 case MENU_NONE: // No transition => do nothing
125 break;
126 case MENU_ZERO:
127 ADC3_IN4_single_init();
128 ADC3_IN4_single_read();
129 break;
130 case MENU_ONE:
131 ADC3_IN4_timer_init();
132 ADC3_IN4_timer_start();
133 break;
134 case MENU_TWO:
135 ADC3_IN4_DMA_init();
136 ADC3_IN4_DMA_start();
137 break;
138 case MENU_THREE:
139 ADC1_IN13_ADC2_IN5_dual_init();
140 ADC1_IN13_ADC2_IN5_dual_start();
141 break;
142 case MENU_FOUR:
143 ADC2_IN13_IN5_scan_init();
144 ADC2_IN13_IN5_scan_start();
145 break;
146 case MENU_FIVE:
147 ADC3_IN13_IN4_scan_init();
148 ADC3_IN13_IN4_scan_start();
149 break;
150 default: // Should never occur
151 break;
152 }
153
154 HAL_Delay(200); // Wait or sleep
155 }
156}

163static void SystemClock_Config(void){
164 RCC_OscInitTypeDef RCC_OscInitStruct = {0};
165 RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};
166 RCC_PeriphCLKInitTypeDef PeriphClkInitStruct = {0};
167 /* Configure the main internal regulator output voltage */
168 __HAL_RCC_PWR_CLK_ENABLE();
169 __HAL_PWR_VOLTAGESCALING_CONFIG(PWR_REGULATOR_VOLTAGE_SCALE1);
170 /* Initialize High Speed External Oscillator and PLL circuits */
171 RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
172 RCC_OscInitStruct.HSEState = RCC_HSE_ON;
173 RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
174 RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
175 RCC_OscInitStruct.PLL.PLLM = 8;
176 RCC_OscInitStruct.PLL.PLLN = 336;
177 RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV2;
178 RCC_OscInitStruct.PLL.PLLQ = 7;
179 HAL_RCC_OscConfig(&RCC_OscInitStruct);
180 /* Initialize gates and clock dividers for CPU, AHB and APB busses */
181 RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_SYSCLK
182 | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2;
183 RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
184 RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
185 RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV4;
186 RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV2;
187 HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_5);
188 /* Initialize PLL and clock divider for the LCD */
189 PeriphClkInitStruct.PeriphClockSelection = RCC_PERIPHCLK_LTDC;
190 PeriphClkInitStruct.PLLSAI.PLLSAIN = 192;
191 PeriphClkInitStruct.PLLSAI.PLLSAIR = 4;
192 PeriphClkInitStruct.PLLSAIDivR = RCC_PLLSAIDIVR_8;
193 HAL_RCCEx_PeriphCLKConfig(&PeriphClkInitStruct);
194 /* Set clock prescaler for ADCs */
195 ADC->CCR |= ADC_CCR_ADCPRE_0;
196}

199/** ***************************************************************************
200 * @brief Disable the GYRO on the microcontroller board.
201 *
202 * @note MISO of the GYRO is connected to PF8 and CS to PC1.
203 * @n Some times the GYRO goes into an undefined mode at startup
204 * and pulls the MISO low or high thus blocking the analog input on PF8.
205 * @n The simplest solution is to pull the CS of the GYRO low for a short while
206 * which is done with the code below.
207 * @n PF8 is also reconfigured.
208 * @n An other solution would be to remove the GYRO
209 * from the microcontroller board by unsoldering it.
210 *****************************************************************************/
211static void gyro_disable(void)
212{
213 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
214 /* Disable PC1 and PF8 first */
215 GPIOC->MODER &= ~GPIO_MODER_MODER1_Msk; // Reset mode for PC1
216 GPIOC->MODER |= 1UL << GPIO_MODER_MODER1_Pos; // Set PC1 as output
217 GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
218 HAL_Delay(10); // Wait some time
219 GPIOC->MODER |= 3UL << GPIO_MODER_MODER1_Pos; // Analog PC1 = ADC123_IN11
220 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
221 GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8_Msk; // Reset speed of PF8
222 GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8_Msk; // Reset alternate function of PF8
223 GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8_Msk; // Reset pulup/down of PF8
224 HAL_Delay(10); // Wait some time
225 GPIOF->MODER |= 3UL << GPIO_MODER_MODER8_Pos; // Analog mode PF8 = ADC3_IN4
226}
