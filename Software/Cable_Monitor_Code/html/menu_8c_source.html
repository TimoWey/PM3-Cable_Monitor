329/** ***************************************************************************
330 * @brief Draw the menu onto the display.
331 *
332 * Each menu entry has two lines.
333 * Text and background colors are applied.
334 * @n These attributes are defined in the variable MENU_draw[].
335 *****************************************************************************/
336void MENU_draw(MENU_type_t type) {
337 BSP_LCD_SetFont(MENU_FONT);
338 uint32_t x, y, m, w, h, offset;
339 y = MENU_Y;
340 m = MENU_MARGIN;
341 w = BSP_LCD_GetXSize() / MENU_NUM_OF_MENU_ELEM;
342 h = MENU_HEIGHT;
343 offset = 0;
344 if (type < MENU_NUM_OF_DIFF_MENU) {
345 offset = type * MENU_NUM_OF_MENU_ELEM; // calculate the menu
346 }
347 for (uint32_t i = 0; i < MENU_NUM_OF_MENU_ELEM; i++) {
348 x = i * w;
349 BSP_LCD_SetTextColor(MENU_entry[i + offset].back_color);
350 BSP_LCD_FillRect(x + m, y + m, w - 2 * m, h - 2 * m);
351 BSP_LCD_SetBackColor(MENU_entry[i + offset].back_color);
352 BSP_LCD_SetTextColor(MENU_entry[i + offset].text_color);
353 BSP_LCD_DisplayStringAt(x + 3 * m, y + 3 * m,
354 (uint8_t*)MENU_entry[i + offset].line1,
355 LEFT_MODE);
356 BSP_LCD_DisplayStringAt(x + 3 * m, y + h / 2,
357 (uint8_t*)MENU_entry[i + offset].line2,
358 LEFT_MODE);
359 }
360}

461/** ***************************************************************************
462 * @brief Check for selection/transition
463 *
464 * If the last transition has been consumed (MENU_NONE == MENU_transition)
465 * and the touchscreen has been touched for a defined period
466 * the variable MENU_transition is set to the touched item.
467 * @note Evalboard revision E (blue PCB) has an inverted y-axis
468 * in the touch controller compared to the display.
469 * Uncomment or comment the <b>\#define EVAL_REV_E</b> in main.h accordingly.
470 *****************************************************************************/
471static void MENU_check_transition(MENU_type_t type) {
472 static MENU_item_t item_old = MENU_NONE;
473 static MENU_item_t item_new = MENU_NONE;
474 static TS_StateTypeDef TS_State; // State of the touch controller
475 uint32_t offset = 0;
476 BSP_TS_GetState(&TS_State); // Get the state
477#ifdef EVAL_REV_E
478 /* Evalboard revision E (blue) has an inverted y-axis in the touch
479 * controller */
480 TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
481#endif
482 if (TS_State.TouchDetected) { // If a touch was detected
483 /* Do only if last transition not pending anymore */
484 if (MENU_NONE == MENU_transition) {
485 item_old = item_new; // Store old item
486 /* If touched within the menu bar? */
487 if ((MENU_Y < TS_State.Y) && (MENU_Y + MENU_HEIGHT > TS_State.Y)) {
488 if (type < MENU_NUM_OF_DIFF_MENU) {
489 offset = type * MENU_NUM_OF_MENU_ELEM; // calculate the
490 // menu
491 }
492 item_new = (TS_State.X // Calculate new item
493 / (BSP_LCD_GetXSize() / MENU_NUM_OF_MENU_ELEM)) +
494 offset;
495 if ((0 > item_new) || (MENU_ENTRY_COUNT <= item_new)) {
496 item_new = MENU_NONE; // Out of bounds
497 }
498 if (item_new == item_old) { // 2 times the same menu item
499 item_new = MENU_NONE;
500 MENU_transition = item_old;
501 }
502 }
503 }
504 }

507/** ***************************************************************************
508 * @brief Interrupt handler for the touchscreen
509 *
510 * @note BSP_TS_ITConfig(); must be called in the main function
511 * to enable touchscreen interrupt.
512 * @note There are timing issues when interrupt is enabled.
513 * It seems that polling is the better choice with this evaluation board.
514 * @n Call MENU_check_transition() from the while loop in main for polling.
515 *
516 * The touchscreen interrupt is connected to PA15.
517 * @n The interrupt handler for external line 15 to 10 is called.
518 *****************************************************************************/
519void EXTI15_10_IRQHandler(void) {
520 if (EXTI->PR & EXTI_PR_PR15) { // Check if interrupt on touchscreen
521 EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
522 if (BSP_TS_ITGetStatus()) { // Get interrupt status
523 BSP_TS_ITClear(); // Clear touchscreen controller int.
524 MENU_check_transition(MENU_NONE);
525 }
526 EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
527 }
