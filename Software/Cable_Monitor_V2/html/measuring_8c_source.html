77/** ***************************************************************************
78 * @brief Initialize ADC, timer and DMA for sequential acquisition = scan mode
79 *
80 * Uses ADC3 and DMA2_Stream1 channel2
81 * @n The ADC3 trigger is set to TIM2 TRGO event
82 * @n At each trigger all inputs are converted sequentially
83 * and transfered to memory by the DMA.
84 * @n The DMA triggers the transfer complete interrupt when all data is ready.
85 * @n The inputs used are ADC123_IN13 = GPIO PC3, ADC3_IN4 = GPIO PF6,
86 * ADC123_IN11 = GPIO PC1 and ADC3_IN6 = GPIO PF8.
87 *****************************************************************************/
88static void MEAS_ADC3_IN13_IN4_IN11_IN6_scan_init(void)
89{
90 __HAL_RCC_ADC3_CLK_ENABLE(); // Enable Clock for ADC3
91 ADC3->SQR1 |= (ADC_SQR1_L_1 | ADC_SQR1_L_0); // Convert 4 inputs
92 ADC3->SQR3 |= (13UL << ADC_SQR3_SQ1_Pos); // Input 13 = first conversion
93 ADC3->SQR3 |= (4UL << ADC_SQR3_SQ2_Pos); // Input 4 = second conversion
94 ADC3->SQR3 |= (11UL << ADC_SQR3_SQ3_Pos); // Input 11 = first conversion
95 ADC3->SQR3 |= (6UL << ADC_SQR3_SQ4_Pos); // Input 6 = second conversion
96 ADC3->CR1 |= ADC_CR1_SCAN; // Enable scan mode
97 ADC3->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
98 ADC3->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
99 ADC3->CR2 |= ADC_CR2_DMA; // Enable DMA mode
100 __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
101 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 1
102 while (DMA2_Stream1->CR & DMA_SxCR_EN) { ; } // Wait for DMA to finish
103 DMA2->LIFCR |= DMA_LIFCR_CTCIF1; // Clear transfer complete interrupt fl.
104 DMA2_Stream1->CR |= (2UL << DMA_SxCR_CHSEL_Pos); // Select channel 2
105 DMA2_Stream1->CR |= DMA_SxCR_PL_1; // Priority high
106 DMA2_Stream1->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
107 DMA2_Stream1->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
108 DMA2_Stream1->CR |= DMA_SxCR_MINC; // Increment memory address pointer
109 DMA2_Stream1->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
110 DMA2_Stream1->NDTR = INPUT_COUNT*ADC_NUMS; // Number of data items to transfer
111 DMA2_Stream1->PAR = (uint32_t)&ADC3->DR; // Peripheral register address
112 DMA2_Stream1->M0AR = (uint32_t)ADC_samples; // Buffer memory loc. address
113}

116/** ***************************************************************************
117 * @brief Start DMA, ADC and timer
118 *
119 *****************************************************************************/
120static void MEAS_ADC3_scan_start(void)
121{
122 DMA2_Stream1->CR |= DMA_SxCR_EN; // Enable DMA
123 NVIC_ClearPendingIRQ(DMA2_Stream1_IRQn); // Clear pending DMA interrupt
124 NVIC_EnableIRQ(DMA2_Stream1_IRQn); // Enable DMA interrupt in the NVIC
125 ADC3->CR2 |= ADC_CR2_ADON; // Enable ADC3
126 TIM2->CR1 |= TIM_CR1_CEN; // Enable timer
127}

130/** ***************************************************************************
131 * @brief Configure GPIOs in analog mode.
132 *
133 * @note The input number for the ADCs is not equal to the GPIO pin number!
134 * - ADC3_IN4 = GPIO PF6
135 * - ADC123_IN13 = GPIO PC3
136 *****************************************************************************/
137void MEAS_GPIO_analog_init(void)
138{
139 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
140 GPIOF->MODER |= (GPIO_MODER_MODER6_Msk);// Analog mode for PF6 = ADC3_IN4
141 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
142 GPIOC->MODER |= (GPIO_MODER_MODER3_Msk);// Analog mode for PC3 = ADC123_IN13
143}

159/** ***************************************************************************
160 * @brief Interrupt handler for DMA2 stream 1
161 *
162 * The samples from the ADC3 have been transfered to memory by the DMA2 Stream1
163 * and are ready for processing.
164 *****************************************************************************/
165void DMA2_Stream1_IRQHandler(void)
166{
167 if (DMA2->LISR & DMA_LISR_TCIF1) { // Stream1 transfer compl. interrupt f.
168 NVIC_DisableIRQ(DMA2_Stream1_IRQn); // Disable DMA interrupt in the NVIC
169 NVIC_ClearPendingIRQ(DMA2_Stream1_IRQn);// Clear pending DMA interrupt
170 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA
171 while (DMA2_Stream1->CR & DMA_SxCR_EN) { ; } // Wait for DMA to finish
172 DMA2->LIFCR |= DMA_LIFCR_CTCIF1;// Clear transfer complete interrupt fl.
173 TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
174 ADC3->CR2 &= ~ADC_CR2_ADON; // Disable ADC3
175 ADC3->CR2 &= ~ADC_CR2_DMA; // Disable DMA mode
176 MEAS_ADC_reset();
177 MEAS_data_ready = true;
178 }
179}

182/** ***************************************************************************
183 * @brief Start the measurement
184 * @return ADC samples pointer
185 *
186 * @note The result is stored alternated e.g. every 4th is together.
187 *****************************************************************************/
188uint32_t *MEAS_start_meas(void)
189{
190 MEAS_Buffer_reset(INPUT_COUNT, ADC_samples);
191 MEAS_ADC3_IN13_IN4_IN11_IN6_scan_init();
192 MEAS_ADC3_scan_start();
193 while(!MEAS_data_ready);
194 MEAS_data_ready = false;
195
196 return ADC_samples;
197}
