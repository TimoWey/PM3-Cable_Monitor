119/** ***************************************************************************
120 * @brief Calculate signal strength in the frequency domain
121 * @param[in] *rfft_input Array of signal
122 * @param[in] fft_len Length of array
123 * @param[in] frequency Amplitude at this frequency
124 * @return Amplitude in 3.3V/4096
125 *
126 * @note To get an optimal result from the FFT, sampling has to be done over a
127 * whole number of signal periods (prevent the so called leakage effect).
128 * E.g. sampling over 100ms spans exactly 5 periods of 50Hz.
129 * @note The FFT needs block sizes of a power of 2. E.g. N = 64 samples.
130 * Combined with Nâˆ™Ts = 100ms defines the
131 * sampling frequency fs = 64/100ms = 640Hz
132 *****************************************************************************/
133static float CALC_get_signalstrength(float *rfft_input, uint8_t fft_len,
134 uint8_t frequency)
135{
136 const uint16_t samplingFreq = MEAS_get_samp_freq();
137 arm_rfft_fast_instance_f32 fft_structer;
138 float rfft_cmplx_output[fft_len];
139 uint8_t fft_index = frequency / (samplingFreq / fft_len);
140 float rfft_mag_output[fft_len];
141
142 arm_rfft_fast_init_f32(&fft_structer, fft_len);
143 arm_rfft_fast_f32(&fft_structer, rfft_input, rfft_cmplx_output, 0);
144 /* Calculate magnitude of complex coefficients (2* at return) */
145 arm_cmplx_mag_f32(rfft_cmplx_output, rfft_mag_output, fft_len / 2);
146
147 return (2 * rfft_mag_output[fft_index] / fft_len);
148}

151/** ***************************************************************************
152 * @brief Calculate angle and distance to board center from one side
153 * @param[in] distance_pad_1 Distance from wire to pad
154 * @param[in] distance_pad_2 Distance from wire to pad
155 * @param[out] *distance_center Distance from wire to board center
156 * @param[out] *angle Angle pointing to wire in degree
157 *
158 * @note Angle always positiv
159 * @bug Skips small angles at short distances
160 *****************************************************************************/
161static void CALC_calc_distance_and_angle_from_one_side(float distance_pad_1,
162 float distance_pad_2, float *distance_center,
163 float *angle)
164{
165 float cos_a; // Ratio needed for calculation
166
167 /* Divide by 5 to compensate that pad signal increases at an angle */
168 cos_a = (POW2(distance_pad_2) + POW2(2 * D_P) - POW2(distance_pad_1)) /
169 (2 * distance_pad_2 * 2 * D_P) / 5;
170 if (cos_a < -1) {
171 cos_a = -1;
172 }
173 if (1 < cos_a) {
174 cos_a = 1;
175 }
176
177 if ((POW2(distance_pad_2) + POW2(D_P) - 2 * distance_pad_2 * D_P * cos_a) <
178 0.0001) {
179 (*distance_center) = 0.0001;
180 } else {
181 (*distance_center) = sqrtf(POW2(distance_pad_2) + POW2(D_P) -
182 2 * distance_pad_2 * D_P * cos_a);
183 }
184
185 (*angle) = distance_pad_2 / (*distance_center) * sinf(acosf(cos_a));
186 if ((*angle) < -1) {
187 (*angle) = -1;
188 }
189 if (1 < (*angle)) {
190 (*angle) = 1;
191 }
192 (*angle) = asinf((*angle)) * (180 / PI) - 90;
193}

196/** ***************************************************************************
197 * @brief Calculate angle and distance to board center
198 * @param[in] distance_pad_l Left distance from wire to pad
199 * @param[in] distance_pad_r Right distance from wire to pad
200 * @param[out] *distance_center Distance from wire to board center
201 * @param[out] *angle Angle pointing to wire in degree
202 *****************************************************************************/
203static void CALC_calc_distance_and_angle(float distance_pad_l,
204 float distance_pad_r, float *distance_center,
205 float *angle)
206{
207 float d1, d2; // Distance from wire to board center
208 float a1, a2; // Angle pointing to wire
209 CALC_calc_distance_and_angle_from_one_side(distance_pad_l, distance_pad_r,
210 &d1, &a1);
211 CALC_calc_distance_and_angle_from_one_side(distance_pad_r, distance_pad_l,
212 &d2, &a2);
213
214 *distance_center = (d1 + d2) / 2;
215 *angle = (a1 + a2) / 2;
216
217 /* Determine in which direction the angle points */
218 if (distance_pad_r < distance_pad_l) {
219 *angle = -*angle;
220 }
221}

224/** ***************************************************************************
225 * @brief Single measurement
226 * @param[out] *meas_data Measurement data @see CALC_meas_data_t
227 * @param[in] phase Select which phase mode for measurement
228 *
229 * @note Standard deviation in data will not be changed.
230 *****************************************************************************/
231void CALC_start_single_measurement(CALC_meas_data_t *meas_data, CALC_phase_item_t phase)
232{
233 uint8_t num_of_chan = MEAS_get_num_of_chan();
234 uint8_t fft_len = MEAS_get_num_of_samples();
235 uint32_t *ADC_samples = MEAS_start_meas();
236
237 meas_data->error = CALC_ERROR_NONE;
238
239 /* Split samples and convert to float */
240 float ADC_samples_pad_r[fft_len]; // IN13 PAD_RIGHT
241 float ADC_samples_pad_l[fft_len]; // IN4 PAD_LEFT
242 float ADC_samples_coil_r[fft_len]; // IN11 COIL_RIGHT
243 float ADC_samples_coil_l[fft_len]; // IN6 COIL_LEFT
244 for (uint8_t i = 0; i < fft_len; i++) {
245 ADC_samples_pad_r[i] = (float)ADC_samples[num_of_chan * i];
246 ADC_samples_pad_l[i] = (float)ADC_samples[num_of_chan * i + 1];
247 ADC_samples_coil_r[i] = (float)ADC_samples[num_of_chan * i + 2];
248 ADC_samples_coil_l[i] = (float)ADC_samples[num_of_chan * i + 3];
249 }
250
251 /* 50Hz filtered signal strength */
252 float signal_pad_r = CALC_get_signalstrength(ADC_samples_pad_r, fft_len, 50);
253 float signal_pad_l = CALC_get_signalstrength(ADC_samples_pad_l, fft_len, 50);
254 float signal_coil_r = CALC_get_signalstrength(ADC_samples_coil_r, fft_len, 50);
255 float signal_coil_l = CALC_get_signalstrength(ADC_samples_coil_l, fft_len, 50);
256
257 /* Approximate distance from wire to pad in mm */
258 float distance_l = a_static_l + b_static_l / signal_pad_r +
259 c_static_l / (POW2(signal_pad_r));
260 float distance_r = a_static_r + b_static_r / signal_pad_l +
261 c_static_r / (POW2(signal_pad_l));
262
263 /* Calculate distance from wire to board center */
264 CALC_calc_distance_and_angle(distance_l, distance_r, &meas_data->distance,
265 &meas_data->angle);
266
267 switch (phase) {
268 case CALC_ONE_PHASE:
269 /* Approximate current with averaging signal */
270 meas_data->current = (signal_coil_r + signal_coil_l) / 2 *
271 meas_data->distance * a_magnetic_1;
272 break;
273 case CALC_THREE_PHASE:
274 /* Approximate current with higher signal */
275 if (signal_coil_r < signal_coil_l) {
276 meas_data->current = signal_coil_l * meas_data->distance * a_magnetic_3;
277 } else {
278 meas_data->current = signal_coil_r * meas_data->distance * a_magnetic_3;
279 }
280 break;
281 }
282
283 if (20 < meas_data->current) {
284 meas_data->error = CALC_ERROR_OVERCURRENT;
285 }
286 if (cal_distance[2] < meas_data->distance) {
287 meas_data->error = CALC_ERROR_DISCONNECT;
288 }
289 if (90 < meas_data->angle) {
290 meas_data->angle = 90;
291 }
292 if (meas_data->angle < -90) {
293 meas_data->angle = -90;
294 }
295}

298/** ***************************************************************************
299 * @brief Accurate measurement
300 * @param[out] *meas_data Measurement data @see CALC_meas_data_t
301 * @param[in] phase Select which phase mode for measurement
302 *****************************************************************************/
303void CALC_start_accurate_measurement(CALC_meas_data_t *meas_data, CALC_phase_item_t phase)
304{
305 float array_distance[ACCURATE_MEASUREMENT_LOOPS];
306 float array_angle[ACCURATE_MEASUREMENT_LOOPS];
307 float array_current[ACCURATE_MEASUREMENT_LOOPS];
308 for (uint8_t i = 0; i < ACCURATE_MEASUREMENT_LOOPS; i++) {
309 CALC_start_single_measurement(meas_data, phase);
310 if (meas_data->error != CALC_ERROR_NONE) {
311 break;
312 }
313 array_distance[i] = meas_data->distance;
314 array_angle[i] = meas_data->angle;
315 array_current[i] = meas_data->current;
316 }
317
318 if (meas_data->error == CALC_ERROR_NONE) {
319 arm_mean_f32(array_distance, ACCURATE_MEASUREMENT_LOOPS, &meas_data->distance);
320 arm_std_f32(array_distance, ACCURATE_MEASUREMENT_LOOPS, &meas_data->dist_sdev);
321 arm_mean_f32(array_angle, ACCURATE_MEASUREMENT_LOOPS, &meas_data->angle);
322 arm_std_f32(array_angle, ACCURATE_MEASUREMENT_LOOPS, &meas_data->angl_sdev);
323 arm_mean_f32(array_current, ACCURATE_MEASUREMENT_LOOPS, &meas_data->current);
324 arm_std_f32(array_current, ACCURATE_MEASUREMENT_LOOPS, &meas_data->curr_sdev);
325
326 if (100 <= meas_data->curr_sdev) {
327 meas_data->error = CALC_ERROR_STANDARD_DEVIATION_TOO_HIGH;
328 }
329 if (100 <= meas_data->dist_sdev) {
330 meas_data->error = CALC_ERROR_STANDARD_DEVIATION_TOO_HIGH;
331 }
332 if (100 <= meas_data->angl_sdev) {
333 meas_data->error = CALC_ERROR_STANDARD_DEVIATION_TOO_HIGH;
334 }
335 }
336}

339/** ***************************************************************************
340 * @brief Pad calibration
341 * @param[in] state_pad_cali State of calibration
342 *
343 * To calibrate the pad (or distance) run this function multiple times
344 * while changing the state of parameter state_pad_cali.
345 *****************************************************************************/
346void CALC_start_pad_calibration(CALC_cali_pad_step_t state_pad_cali)
347{
348 float signal_pad_l = 0;
349 float signal_pad_r = 0;
350 static float d1;
351 static float s1_l;
352 static float s1_r;
353 static float d2;
354 static float s2_l;
355 static float s2_r;
356 static float d3;
357 static float s3_l;
358 static float s3_r;
359
360 uint8_t num_of_chan = MEAS_get_num_of_chan();
361 uint8_t fft_len = MEAS_get_num_of_samples();
362
363 if (state_pad_cali < CALC_CALI_P_FINISH) {
364 for (uint8_t i = 0; i < ACCURATE_MEASUREMENT_LOOPS; i++) {
365 uint32_t* ADC_samples = MEAS_start_meas();
366
367 /* Split samples and convert to float */
368 float ADC_samples_pad_r[fft_len]; // IN13 PAD_RIGHT
369 float ADC_samples_pad_l[fft_len]; // IN4 PAD_LEFT
370 for (uint8_t i = 0; i < fft_len; i++) {
371 ADC_samples_pad_r[i] = (float)ADC_samples[num_of_chan * i];
372 ADC_samples_pad_l[i] = (float)ADC_samples[num_of_chan * i + 1];
373 }
374
375 signal_pad_l += CALC_get_signalstrength(ADC_samples_pad_r, fft_len, 50);
376 signal_pad_r += CALC_get_signalstrength(ADC_samples_pad_l, fft_len, 50);
377 }
378 signal_pad_l /= ACCURATE_MEASUREMENT_LOOPS;
379 signal_pad_r /= ACCURATE_MEASUREMENT_LOOPS;
380 }
381
382 if (CALC_CALI_P_FIRST == state_pad_cali) {
383 /* Calculate the wire to pad distance */
384 d1 = sqrtf(POW2(cal_distance[CALC_CALI_P_FIRST - 1]) + POW2(D_P));
385 s1_l = signal_pad_l;
386 s1_r = signal_pad_r;
387 } else if (CALC_CALI_P_SECOND == state_pad_cali) {
388 /* Calculate the wire to pad distance */
389 d2 = sqrtf(POW2(cal_distance[CALC_CALI_P_SECOND - 1]) + POW2(D_P));
390 s2_l = signal_pad_l;
391 s2_r = signal_pad_r;
392
393 } else if (CALC_CALI_P_THIRD == state_pad_cali) {
394 /* Calculate the wire to pad distance */
395 d3 = sqrtf(POW2(cal_distance[CALC_CALI_P_THIRD - 1]) + POW2(D_P));
396 s3_l = signal_pad_l;
397 s3_r = signal_pad_r;
398
399 } else if (CALC_CALI_P_FINISH == state_pad_cali) {
400 /* Calculate coefficients for distance approximation */
401 a_static_l = ((d1 * POW2(s1_l) * (s2_l - s3_l) -
402 d2 * (s1_l - s3_l) * POW2(s2_l) +
403 d3 * (s1_l - s2_l) * POW2(s3_l)) /
404 ((POW2(s1_l) - s1_l * (s2_l + s3_l) + s2_l * s3_l) *
405 (s2_l - s3_l)));
406 b_static_l = ((-(d1 * POW2(s1_l) * (POW2(s2_l) - POW2(s3_l)) -
407 d2 * (POW2(s1_l) - POW2(s3_l)) * POW2(s2_l) +
408 d3 * (POW2(s1_l) - POW2(s2_l)) * POW2(s3_l))) /
409 ((POW2(s1_l) - s1_l * (s2_l + s3_l) + s2_l * s3_l) *
410 (s2_l - s3_l)));
411 c_static_l = (((d1 * s1_l * (s2_l - s3_l) - d2 * (s1_l - s3_l) * s2_l +
412 d3 * (s1_l - s2_l) * s3_l) *
413 s1_l * s2_l * s3_l) /
414 ((POW2(s1_l) - s1_l * (s2_l + s3_l) + s2_l * s3_l) *
415 (s2_l - s3_l)));
416
417 a_static_r = ((d1 * POW2(s1_r) * (s2_r - s3_r) -
418 d2 * (s1_r - s3_r) * POW2(s2_r) +
419 d3 * (s1_r - s2_r) * POW2(s3_r)) /
420 ((POW2(s1_r) - s1_r * (s2_r + s3_r) + s2_r * s3_r) *
421 (s2_r - s3_r)));
422 b_static_r = ((-(d1 * POW2(s1_r) * (POW2(s2_r) - POW2(s3_r)) -
423 d2 * (POW2(s1_r) - POW2(s3_r)) * POW2(s2_r) +
424 d3 * (POW2(s1_r) - POW2(s2_r)) * POW2(s3_r))) /
425 ((POW2(s1_r) - s1_r * (s2_r + s3_r) + s2_r * s3_r) *
426 (s2_r - s3_r)));
427 c_static_r = (((d1 * s1_r * (s2_r - s3_r) - d2 * (s1_r - s3_r) * s2_r +
428 d3 * (s1_r - s2_r) * s3_r) *
429 s1_r * s2_r * s3_r) /
430 ((POW2(s1_r) - s1_r * (s2_r + s3_r) + s2_r * s3_r) *
431 (s2_r - s3_r)));
432 }
433}

436/** ***************************************************************************
437 * @brief Coil calibration
438 * @param[in] state_phase Select which phase mode for measurement
439 * @param[in] state_coil_cali State of calibration
440 *
441 * To calibrate the coil (or current) run this function multiple times
442 * while changing the state of parameter state_coil_cali.
443 *****************************************************************************/
444void CALC_start_coil_calibration(CALC_phase_item_t state_phase,
445 CALC_cali_coil_step_t state_coil_cali)
446{
447 static CALC_meas_data_t meas_data;
448 static float signal;
449
450 if (CALC_CALI_C_FIRST == state_coil_cali) {
451 CALC_start_accurate_measurement(&meas_data, state_phase);
452 if (state_phase == CALC_ONE_PHASE) {
453 signal = meas_data.current / (a_magnetic_1 * meas_data.distance);
454 } else if (state_phase == CALC_THREE_PHASE) {
455 signal = meas_data.current / (a_magnetic_3 * meas_data.distance);
456 }
457
458 } else if (CALC_CALI_C_FINISH == state_coil_cali) {
459 if (state_phase == CALC_ONE_PHASE) {
460 a_magnetic_1 =
461 cal_current[CALC_CALI_C_FIRST - 1] / (signal * meas_data.distance);
462 } else if (state_phase == CALC_THREE_PHASE) {
463 a_magnetic_3 =
464 cal_current[CALC_CALI_C_FIRST - 1] / (signal * meas_data.distance);
465 }
466 }
467}
