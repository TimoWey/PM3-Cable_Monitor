110static void MAIN_system_clock_config(void)
111{
112 RCC_OscInitTypeDef RCC_OscInitStruct = {0};
113 RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};
114 RCC_PeriphCLKInitTypeDef PeriphClkInitStruct = {0};
115
116 /* Configure the main internal regulator output voltage. */
117 __HAL_RCC_PWR_CLK_ENABLE();
118 __HAL_PWR_VOLTAGESCALING_CONFIG(PWR_REGULATOR_VOLTAGE_SCALE1);
119
120 /* Initialize High Speed External Oscillator and PLL circuits. */
121 RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
122 RCC_OscInitStruct.HSEState = RCC_HSE_ON;
123 RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
124 RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
125 RCC_OscInitStruct.PLL.PLLM = 8;
126 RCC_OscInitStruct.PLL.PLLN = 336;
127 RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV2;
128 RCC_OscInitStruct.PLL.PLLQ = 7;
129 HAL_RCC_OscConfig(&RCC_OscInitStruct);
130
131 /* Initialize gates and clock dividers for CPU, AHB and APB busses. */
132 RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_SYSCLK
133 | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2;
134 RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
135 RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
136 RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV4;
137 RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV2;
138 HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_5);

141 PeriphClkInitStruct.PeriphClockSelection = RCC_PERIPHCLK_LTDC;
142 PeriphClkInitStruct.PLLSAI.PLLSAIN = 192;

144 PeriphClkInitStruct.PLLSAIDivR = RCC_PLLSAIDIVR_8;
145 HAL_RCCEx_PeriphCLKConfig(&PeriphClkInitStruct);

147 /* Set clock prescaler for ADCs. */

152/** ***************************************************************************
153 * @brief Disable the GYRO on the microcontroller board
154 *
155 * Disable GYRO on PC1 and PF8 and set them as an analog input.
156 * @note MISO of the GYRO is connected to PF8 and CS to PC1.
157 * @n Some times the GYRO goes into an undefined mode at startup
158 * and pulls the MISO low or high thus blocking the analog input on PF8.
159 * @n The simplest solution is to pull the CS of the GYRO low for a short while
160 * which is done with the code below.
161 * @n PF8 is also reconfigured.
162 * @n An other solution would be to remove the GYRO
163 * from the microcontroller board by unsoldering it.
164 *****************************************************************************/
165static void MAIN_gyro_disable(void)
166{
167 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
168 GPIOC->MODER &= ~GPIO_MODER_MODER1; // Reset mode for PC1
169 GPIOC->MODER |= GPIO_MODER_MODER1_0; // Set PC1 as output
170 GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
171 HAL_Delay(10); // Wait some time
172 GPIOC->MODER |= GPIO_MODER_MODER1_Msk; // Analog mode PC1 = ADC123_IN11
173
174 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
175 GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8; // Reset speed of PF8
176 GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8; // Reset alternate func. of PF8
177 GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8; // Reset pull up/down of PF8
178 HAL_Delay(10); // Wait some time
179 GPIOF->MODER |= GPIO_MODER_MODER8_Msk; // Analog mode for PF8 = ADC3_IN4
180}
