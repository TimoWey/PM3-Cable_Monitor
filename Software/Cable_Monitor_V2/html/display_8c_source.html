142/** ***************************************************************************
143 * @brief Displays the angle screen
144 * @param[in] angle Angle that is displayed
145 * @param[in] y_offset Y-position on the screen
146 *
147 * @note At the Y-position is the beginning of the angle display.
148 *****************************************************************************/
149static void DISP_angle_screen(float angle, uint16_t y_offset)
150{
151 const uint16_t X_POS = 120; //x position of the block
152 const uint16_t RAD = 70; //radius position of the block
153
154 /* calculate the angle from degree to radian */
155 angle = angle/180*M_PI;
156
157 /* Set font correctly */
158 BSP_LCD_SetFont(&Font16);
159 BSP_LCD_SetBackColor(LCD_COLOR_WHITE);
160 BSP_LCD_SetTextColor(LCD_COLOR_BLACK);
161
162 /* display the angle */
163 BSP_LCD_DrawCircle(X_POS, y_offset, RAD);
164 BSP_LCD_SetTextColor(LCD_COLOR_WHITE);
165 BSP_LCD_FillRect(0, y_offset, 240, RAD+1);
166 BSP_LCD_SetTextColor(LCD_COLOR_BLACK);
167 BSP_LCD_FillCircle(X_POS, y_offset, RAD/14);
168 BSP_LCD_DrawHLine(X_POS-RAD, y_offset, 2*RAD);
169 BSP_LCD_DrawLine(X_POS, y_offset, X_POS+RAD*sinf(angle),
170 y_offset-RAD*cosf(angle));
171}

174/** ***************************************************************************
175 * @brief Displays the curves screen
176 * @param[in] y_offset Y-position on the screen
177 * @param[in] *samples Array with the values
178 * @param[in] diff_signals Number of different signals
179 * @param[in] num_of_sam Number of samples
180 *
181 * @note The different curves are shown with different colours
182 * @n on the display at the Y-position.
183 *****************************************************************************/
184static void DISP_curves_screen(uint16_t y_offset, uint32_t *samples,
185 uint8_t diff_signals, uint8_t num_of_sam)
186{
187 const char *curve_name[4]={"PR", "PL", "CR", "CL"};
188 const uint32_t color[4]={LCD_COLOR_DARKCYAN, LCD_COLOR_BLUE,
189 LCD_COLOR_MAGENTA, LCD_COLOR_RED};
190 const uint8_t ADC_RES = 12; // Resolution
191 const uint8_t MAX_NUMBER_OF_SAMPLES = 60;
192 const uint8_t SIGNAL_AMP = 65;
193 const uint32_t f = (1 << ADC_RES) / SIGNAL_AMP + 1; // Scaling factor
194 uint32_t data;
195 uint32_t data_last;
196 if(num_of_sam > MAX_NUMBER_OF_SAMPLES) {
197 num_of_sam = MAX_NUMBER_OF_SAMPLES;
198 }
199 /* Draw the different curves */
200 for (uint8_t j = 0; j < diff_signals; j++) {
201 BSP_LCD_SetTextColor(color[j]);
202 BSP_LCD_DisplayStringAt(135+25*j, y_offset-79,
203 (uint8_t *)curve_name[j], LEFT_MODE);
204 data = samples[j] / f;
205 for (uint8_t i = 1; i < num_of_sam; i++){
206 data_last = data;
207 data = (samples[diff_signals*i+j]) / f;
208 /* Limit value, prevent crash */
209 if (data > y_offset) { data = y_offset; }
210 BSP_LCD_DrawLine(4*(i-1), y_offset-data_last, 4*i, y_offset-data);
211 }
212 }
213}
