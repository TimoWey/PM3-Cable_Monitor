134/** ***************************************************************************
135 * @brief Check for selection/transition
136 * @param[in] type Actual menu type
137 *
138 * If the last transition has been consumed (MENU_NONE == MENU_transition)
139 * @n and the touchscreen has been touched for a defined period
140 * @n the variable MENU_transition is set to the touched item.
141 * @note Evalboard revision E (blue PCB) has an inverted y-axis
142 * in the touch controller compared to the display.
143 * @n Uncomment or comment the <b>\#define EVAL_REV_E</b> in main.h accordingly.
144 *****************************************************************************/
145static void MENU_check_transition(MENU_type_t type)
146{
147 static MENU_item_t item_old = MENU_NONE;
148 static MENU_item_t item_new = MENU_NONE;
149 static TS_StateTypeDef TS_State; // State of the touch controller
150 uint32_t offset = 0;
151 BSP_TS_GetState(&TS_State); // Get the state
152#ifdef EVAL_REV_E
153/* Evalboard revision E (blue) has an inverted y-axis in the touch controller */
154 TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
155#endif
156 if (TS_State.TouchDetected) { // If a touch was detected
157 /* Do only if last transition not pending anymore */
158 if (MENU_NONE == MENU_transition) {
159 item_old = item_new; // Store old item
160 /* If touched within the menu bar? */
161 if ((MENU_Y < TS_State.Y) && (MENU_Y+MENU_HEIGHT > TS_State.Y)) {
162 if(type < MENU_NUM_OF_DIFF_MENU) {
163 offset = type*MENU_NUM_OF_MENU_ELEM; //calculate the menu
164 }
165 item_new = (TS_State.X // Calculate new item
166 / (BSP_LCD_GetXSize()/MENU_NUM_OF_MENU_ELEM)) + offset;
167 if ((0 > item_new) || (MENU_ENTRY_COUNT <= item_new)) {
168 item_new = MENU_NONE; // Out of bounds
169 }
170 if (item_new == item_old) { // 2 times the same menu item
171 item_new = MENU_NONE;
172 MENU_transition = item_old;
173 }
174 }
175 }
176 }

180/** ***************************************************************************
181 * @brief Calibration menu for the distance
182 * @param[in] command Actual menu item
183 * @return Actual calibration step for the distance
184 *
185 * Controls the calibration of the distance.
186 *****************************************************************************/
187static uint8_t MENU_pad_cali(MENU_item_t command)
188{
189 static CALC_cali_pad_step_t state_pad_cali = CALC_CALI_P_START;
190
191 if (state_pad_cali >= CALC_CALI_P_END) {
192 state_pad_cali = CALC_CALI_P_START;
193 }
194 if (state_pad_cali < CALC_CALI_P_FINISH) {
195 switch (command) {
196 case MENU_P_NE:
197 state_pad_cali++;
198 break;
199 case MENU_P_AB:
200 state_pad_cali = CALC_CALI_P_ABOARD;
201 break;
202 case MENU_P_BA:
203 if (state_pad_cali != CALC_CALI_P_START) {
204 state_pad_cali--;
205 }
206 break;
207 default: // Should never occur
208 break;
209 }
210 } else {
211 state_pad_cali++;
212 }
213
214 if (command != MENU_P_BA) {
215 CALC_start_pad_calibration(state_pad_cali);
216 }
217 DISP_pad_calibration(state_pad_cali);
218 if (state_pad_cali == CALC_CALI_P_FINISH) {
219 state_pad_cali++;
220 }
221 return state_pad_cali;
222}

225/** ***************************************************************************
226 * @brief Calibration menu for the current
227 * @param[in] command Actual menu item
228 * @return Actual calibration step for the current
229 *
230 * Controls the calibration of the current.
231 *****************************************************************************/
232static uint8_t MENU_coil_cali(MENU_item_t command)
233{
234 static CALC_cali_coil_step_t state_coil_cali = CALC_CALI_C_START;
235 static CALC_phase_item_t state_phase = CALC_ONE_PHASE;
236
237 if (state_coil_cali >= CALC_CALI_C_END) {
238 state_coil_cali = CALC_CALI_C_START;
239 }
240 if (state_coil_cali < CALC_CALI_C_FINISH) {
241 switch (command) {
242 case MENU_C_1P:
243 state_coil_cali++;
244 if (state_coil_cali == CALC_CALI_C_FIRST) {
245 state_phase = CALC_ONE_PHASE;
246 }
247 break;
248 case MENU_C_3P:
249 state_coil_cali++;
250 if (state_coil_cali == CALC_CALI_C_FIRST) {
251 state_phase = CALC_THREE_PHASE;
252 }
253 break;
254 case MENU_C_AB:
255 state_coil_cali = CALC_CALI_C_ABOARD;
256 break;
257 default: // Should never occur
258 break;
259 }
260 } else {
261 state_coil_cali++;
262 }
263
264 CALC_start_coil_calibration(state_phase, state_coil_cali);
265 DISP_coil_calibration(state_coil_cali);
266 if (state_coil_cali == CALC_CALI_C_FINISH) {
267 state_coil_cali++;
268 }
269 return state_coil_cali;
270}

273/** ***************************************************************************
274 * @brief Draw the menu on to the display.
275 * @param[in] type Actual menu type
276 *
277 * Each menu entry has two lines.
278 * Text and background colors are applied.
279 * @n These attributes are defined in the variable MENU_draw[].
280 *****************************************************************************/
281void MENU_draw(MENU_type_t type)
282{
283 BSP_LCD_SetFont(MENU_FONT);
284 uint32_t x, y, m, w, h, offset;
285 y = MENU_Y;
286 m = MENU_MARGIN;
287 w = BSP_LCD_GetXSize()/MENU_NUM_OF_MENU_ELEM;
288 h = MENU_HEIGHT;
289 offset = 0;
290 if (type < MENU_NUM_OF_DIFF_MENU) {
291 offset = type*MENU_NUM_OF_MENU_ELEM; //calculate the menu
292 }
293 for (uint32_t i = 0; i < MENU_NUM_OF_MENU_ELEM; i++) {
294 x = i*w;
295 BSP_LCD_SetTextColor(MENU_entry[i+offset].back_color);
296 BSP_LCD_FillRect(x+m, y+m, w-2*m, h-2*m);
297 BSP_LCD_SetBackColor(MENU_entry[i+offset].back_color);
298 BSP_LCD_SetTextColor(MENU_entry[i+offset].text_color);
299 BSP_LCD_DisplayStringAt(x+3*m, y+3*m,
300 (uint8_t *)MENU_entry[i+offset].line1, LEFT_MODE);
301 BSP_LCD_DisplayStringAt(x+3*m, y+h/2,
302 (uint8_t *)MENU_entry[i+offset].line2, LEFT_MODE);
303 }
304}

313void MENU_main(void)
314{
315 static CALC_meas_data_t meas_data;
316 static MENU_type_t menu = MENU_HOME;
317
318 MENU_check_transition(menu);
319
320 switch (MENU_get_transition()) { // Handle user menu choice
321 case MENU_NONE: // No transition => do nothing
322 break;
323
324 case MENU_SING:
325 menu = MENU_M_SI;
326 DISP_info_screen(menu); // Show single meas. menu
327 break;
328 case MENU_ACCU:
329 menu = MENU_M_AC;
330 DISP_info_screen(menu); // Show accurate meas. menu
331 break;
332 case MENU_CALB:
333 menu = MENU_CALI;
334 DISP_info_screen(menu); // Show calibration menu
335 break;
336
337 case MENU_S_1P:
338 CALC_start_single_measurement(&meas_data, CALC_ONE_PHASE);
339 DISP_value_screen_sing(meas_data, CALC_ONE_PHASE);
340 break;
341 case MENU_S_3P:
342 CALC_start_single_measurement(&meas_data, CALC_THREE_PHASE);
343 DISP_value_screen_sing(meas_data, CALC_THREE_PHASE);
344 break;
345
346 case MENU_A_1P:
347 CALC_start_accurate_measurement(&meas_data, CALC_ONE_PHASE);
348 DISP_value_screen_accu(meas_data, CALC_ONE_PHASE);
349 break;
350 case MENU_A_3P:
351 CALC_start_accurate_measurement(&meas_data, CALC_THREE_PHASE);
352 DISP_value_screen_accu(meas_data, CALC_THREE_PHASE);
353 break;
354
355 case MENU_C_PD:
356 menu = MENU_C_PA;
357 DISP_info_screen(menu); // Show distance calibration menu
358 break;
359 case MENU_C_CI:
360 menu = MENU_C_CO;

362 break;
