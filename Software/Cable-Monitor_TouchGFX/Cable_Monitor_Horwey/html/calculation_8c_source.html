133FFT calculate_freq_and_signalstrength(float32_t input_samples[])
134{
135 // remove DC component
136 for (uint32_t i = 0; i < SAMPLE_LEN; i++) {
137 input_samples[i] -= 2047.5;
138 }
139 // Create an array for the FFT output
140 float32_t fft_complex_output[2 * SAMPLE_LEN];
141 // Create an instance of the FFT structure
142 arm_rfft_fast_instance_f32 fft_struct;
143 // Initialize the FFT structure
144 arm_rfft_fast_init_f32(&fft_struct, SAMPLE_LEN);
145 // Perform the FFT on the input samples
146 arm_rfft_fast_f32(&fft_struct, input_samples, fft_complex_output, 0);
147 // Calculate magnitude of complex numbers
148 arm_cmplx_mag_f32(fft_complex_output, fft_complex_output, SAMPLE_LEN / 2);
149 // Find the index of the maximum magnitude in the first half of the array
150 uint32_t maxIndex;
151 float32_t maxValue;
152 arm_max_f32(fft_complex_output, SAMPLE_LEN / 2, &maxValue, &maxIndex);
153 // Create an instance of the FFT structure
154 FFT fft;
155 // Calculate the main frequency
156 float main_frequency =
157 (float)maxIndex * ((float)MEAS_get_samp_freq()) / SAMPLE_LEN;
158 uint8_t given_frequency = 0;
159 // Check if the main frequency is in the range of 45-55 Hz
160 if (main_frequency <= 55 && main_frequency >= 45)
161 {
162 // If yes, round the given frequency to 50 Hz
163 given_frequency = 50;
164 }
165 else if (main_frequency <= 65 && main_frequency >= 55)
166 {
167 // If yes, round the given frequency to 60 Hz
168 given_frequency = 60;
169 }
170 else
171 given_frequency = 0;
172 // calculate the index of the given frequency
173 uint32_t index = (uint32_t)given_frequency * SAMPLE_LEN / MEAS_get_samp_freq();
174 // set the signal strength to 0
175 float signal_strength = 0;
176 // check if the given frequency is 50 or 60 Hz
177 if (given_frequency != 0)
178 {
179 // calculate the signal strength
180 signal_strength = 2 * fft_complex_output[index] / SAMPLE_LEN;
181 fft.error = CALC_ERROR_NONE;
182 }
183 else
184 fft.error = CALC_ERROR_FREQUENCY; // if the given frequency is not 50 or 60 Hz
185 // return the calculated values
186 fft.main_freq = main_frequency;
187 fft.signal_strength = signal_strength;
188 return fft;
189}

272SINGLE_MEAS single_measurement(uint8_t Phase)
273{
274 // Start the measurement
275 uint32_t* samples = MEAS_start_measure();
276 // Create an instance of the SINGLE_MEAS structure
277 SINGLE_MEAS single_meas;
278 // Create an instance of the FFT structure
279 FFT fft;
280
281 // Set the error to none
282 single_meas.error = CALC_ERROR_NONE;
283
284 static float32_t samples_PR[SAMPLE_LEN];
285 static float32_t samples_PL[SAMPLE_LEN];
286 static float32_t samples_HSR[SAMPLE_LEN];
287 static float32_t samples_HSL[SAMPLE_LEN];
288
289 float32_t current_l = 0;
290 float32_t current_r = 0;
291
292 // Convert the samples to the specific channel
293 for (uint32_t i = 0; i < 64; i++)
294 {
295 samples_PR[i] = (float32_t)samples[(i * CHANNEL_NUM)];
296 samples_PL[i] = (float32_t)samples[(i * CHANNEL_NUM) + 1];
297 samples_HSR[i] = (float32_t)samples[(i * CHANNEL_NUM) + 2];
298 samples_HSL[i] = (float32_t)samples[(i * CHANNEL_NUM) + 3];
299 }
300 // Calculate the Signal Strength and the main frequency for each channel using FFT
301 fft = calculate_freq_and_signalstrength(samples_PR);
302 float32_t signal_strength_PR = fft.signal_strength;
303 float32_t main_frequency_PR = fft.main_freq;
304 fft = calculate_freq_and_signalstrength(samples_PL);
305 float32_t signal_strength_PL = fft.signal_strength;
306 float32_t main_frequency_PL = fft.main_freq;
307 fft = calculate_freq_and_signalstrength(samples_HSR);
308 float32_t signal_strength_HSR = fft.signal_strength;
309 fft = calculate_freq_and_signalstrength(samples_HSL);
310 float32_t signal_strength_HSL = fft.signal_strength;
311 // Calculate the main frequency
312 single_meas.frequency = (main_frequency_PR + main_frequency_PL) / 2;
313 // Calculate distance and angle
314 DISTANCE_ANGLE dist_angle = calculate_distance_and_angle(signal_strength_PR, signal_strength_PL);
315 single_meas.distance = dist_angle.distance;
316 single_meas.angle = dist_angle.angle;
317
318 // Calculate the current using the magnetic coefficients for 5A
319 if(Phase == SINGLE_PHASE)
320 {
321 current_l = a_magn2_l_1P * single_meas.distance * signal_strength_HSL;
322 current_r = a_magn2_r_1P * single_meas.distance * signal_strength_HSR;
323 single_meas.current = (current_l + current_r) / 2;
324 }
325 else if(Phase == THREE_PHASE)
326 {
327 if (signal_strength_HSL > signal_strength_HSR)
328 single_meas.current = a_magn2_l_3P * single_meas.distance * signal_strength_HSL;
329 else
330 single_meas.current = a_magn2_r_3P * single_meas.distance * signal_strength_HSR;
331 }
332
333 // Check if the current is smaller than 4 A -> use the magnetic coefficients for 1.2 A
334 if (single_meas.current < 4)
335 {
336 if(Phase == SINGLE_PHASE)
337 {
338 current_l = a_magn1_l_1P * single_meas.distance * signal_strength_HSL;
339 current_r = a_magn1_r_1P * single_meas.distance * signal_strength_HSR;
340 single_meas.current = (current_l + current_r) / 2;
341 }
342 else if(Phase == THREE_PHASE)
343 {
344 if (signal_strength_HSL > signal_strength_HSR)
345 single_meas.current = a_magn1_l_3P * single_meas.distance * signal_strength_HSL;
346 else
347 single_meas.current = a_magn1_r_3P * single_meas.distance * signal_strength_HSR;
348 }
349 }
350
351 // ERROR TESTING:
352
353 // Check if distance is close enough for current calculation and if the current is out of the range of 0-10 A
354 if (single_meas.distance > 30 || (single_meas.current <= 0 && single_meas.current >= 10))
355 single_meas.error = CALC_ERROR_TOO_FAR_AWAY;
356
357 // Check if the current is larger than 10 A -> CALL ERROR OVERCURRENT
358 if (single_meas.current > 10)
359 single_meas.error = CALC_ERROR_OVERCURRENT;
360
361 // Check if the distance is larger than 300 mm or smaller than 0 -> CALL ERROR DISCONNECT
362 if (single_meas.distance > 250)
363 {
364 single_meas.error = CALC_ERROR_DISCONNECT;
365 single_meas.angle = 0;
366 }
367
368 // Check if the frequency is 0 -> CALL ERROR FREQUENCY
369 if (single_meas.frequency == 0)
370 single_meas.error = CALC_ERROR_FREQUENCY;
371
372 // OPTIMIZE:
373 // Limit the angle to +/- 90 degrees
374 if (single_meas.angle > 90)
375 single_meas.angle = 90;
376 else if (single_meas.angle < -90)
377 single_meas.angle = -90;
378
379 // BUZZER LED
380 if(single_meas.distance > 0 && single_meas.distance < 200 && (State_BUZZER == false))
381 TIM14->CCR1 = 50 + (single_meas.distance/4);
382 else
383 TIM14->CCR1 = 0;
384
385 if(single_meas.distance > 0 && single_meas.distance < 200 && (State_LED == false))
386 TIM5->CCR1 = abs(100 - (int)(single_meas.distance/2));
387 else
388 TIM5->CCR1 = 0;
389 // Return the calculated values
390 return single_meas;
391}

404ACCU_MEAS accurate_measurement(uint8_t Phase)
405{
406 // Create an instance of the ACCU_MEAS structure
407 ACCU_MEAS accu_meas;
408
409 // set the error to none
410 accu_meas.error_accu = CALC_ERROR_NONE;
411
412 static float32_t distance[ACCURATE_MEASUREMENT_LOOPS];
413 static float32_t angle[ACCURATE_MEASUREMENT_LOOPS];
414 static float32_t frequency[ACCURATE_MEASUREMENT_LOOPS];

416

423

425 if (single_meas.error != CALC_ERROR_NONE){
426 accu_meas.error_single = single_meas.error;
427 }
428 }
429 // Calculate the mean value of the distance
430 arm_mean_f32(distance, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.distance);
431 // Calculate standard deviation of the distance
432 arm_std_f32(distance, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.distance_std_dev);
433 // Calculate the mean value of the angle
434 arm_mean_f32(angle, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.angle);
435 // Calculate standard deviation of the angle
436 arm_std_f32(angle, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.angle_std_dev);
437 // Calculate the mean value of the frequency
438 arm_mean_f32(frequency, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.frequency);
439 // Calculate standard deviation of the frequency
440 arm_std_f32(frequency, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.frequency_std_dev);
441 // Calculate the mean value of the current
442 arm_mean_f32(current, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.current);
443 // Calculate standard deviation of the current
444 arm_std_f32(current, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.current_std_dev);
445
446 // Check if the standard deviation is larger than 100
447 if (accu_meas.distance_std_dev > 100 || accu_meas.angle_std_dev > 100 || \
448 accu_meas.frequency_std_dev > 100 || accu_meas.current_std_dev > 100)
449 accu_meas.error_accu = CALC_ERROR_DEVIATION_TOO_HIGH;
450 else
451 accu_meas.error_accu = CALC_ERROR_NONE;
452
453 // Return the calculated values
454 return accu_meas;
455}

468DISTANCE_ANGLE calculate_distance_and_angle(float32_t signal_strength_r, float32_t signal_strength_l)
469{
470
471 // Create an instance of the DISTANCE_ANGLE structure
472 DISTANCE_ANGLE dist_angle;
473
474 // Calculate the distance using the coefficients and the actual signal strength
475 float32_t calc_distance = a_r + (b_r / signal_strength_r) + (c_r / POW2(signal_strength_r));
476 // Set the calculated values
477 dist_angle.distance_r = calc_distance;
478
479 // Calculate the distance using the coefficients and the actual signal strength
480 calc_distance = a_l + (b_l / signal_strength_l) + (c_l / POW2(signal_strength_l));
481 // Set the calculated values
482 dist_angle.distance_l = calc_distance;
483
484 // Calculate the distance
485 dist_angle.distance = (dist_angle.distance_l + dist_angle.distance_r) / 2;
486
487 // Calculate the angle using the trigonometry
488 dist_angle.angle = atanf((dist_angle.distance_l - dist_angle.distance_r) / (2 * D_P)) * (180 / PI);
489
490 // Return the calculated values
491 return dist_angle;
492}

545void calculate_coefficients_single_pad(float32_t s[], float32_t d[], float32_t* a, float32_t* b, float32_t* c)
546{
547 // s = x_0, x_1, x_2 d= y_0, y_1, y_2
548 // Calculate the coefficients for the distance approximation from an approximation based on a second degree polynomial
549 // a=((x_0^(2)*(x_1-x_2)*y_0-x_0*(x_1^(2)*y_1-x_2^(2)*y_2)+x_1*(x_1*y_1-x_2*y_2)*x_2)/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
550 // -> Result of Nspire CAS
551
552 *a = (POW2(s[0]) * (s[1] - s[2]) * d[0] - s[0] * (POW2(s[1]) * d[1] - POW2(s[2]) * d[2]) + s[1] * (s[1] * d[1] - s[2] * d[2]) * s[2])
553 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
554 // b = ((−(x_0^(2)*(x_1^(2)*(y_0-y_1)-x_2^(2)*(y_0-y_2))+x_1^(2)*x_2^(2)*(y_1-y_2)))/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
555 // -> Result of Nspire CAS
556
557 *b = (-(POW2(s[0]) * (POW2(s[1]) * (d[0] - d[1]) - POW2(s[2]) * (d[0] - d[2])) + POW2(s[1]) * POW2(s[2]) * (d[1] - d[2])))
558 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
559
560 // c=((x_0*(x_0*(x_1*(y_0-y_1)-x_2*(y_0-y_2))+x_1*x_2*(y_1-y_2))*x_1*x_2)/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
561 // -> Result of Nspire CAS
562 *c = ((s[0] * (s[0] * (s[1] * (d[0] - d[1]) - s[2] * (d[0] - d[2])) + s[1] * s[2] * (d[1] - d[2])) * s[1] * s[2]))
563 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
564}
