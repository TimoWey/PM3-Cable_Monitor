161void MEAS_ADC3_scan_init(void)
162{
163 __HAL_RCC_ADC3_CLK_ENABLE(); // Enable Clock for ADC3
164 ADC3->SQR1 |= (3UL << ADC_SQR1_L_Pos); // Convert 4 inputs (4-1)
165 ADC3->SQR3 |= (4UL << ADC_SQR3_SQ1_Pos); // Input 4 = first conversion
166 ADC3->SQR3 |= (13UL << ADC_SQR3_SQ2_Pos); // Input 13 = second conversion
167 ADC3->SQR3 |= (6UL << ADC_SQR3_SQ3_Pos); // Input 6 = third conversion
168 ADC3->SQR3 |= (11UL << ADC_SQR3_SQ4_Pos); // Input 11 = fourth conversion
169 ADC3->CR1 |= ADC_CR1_SCAN; // Enable scan mode
170 ADC3->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
171 ADC3->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
172 ADC3->CR2 |= ADC_CR2_DMA; // Enable DMA mode
173 __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
174 // set DMA to highest priority
175 NVIC_SetPriority(DMA2_Stream1_IRQn, 0);
176 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 1
177 while (DMA2_Stream1->CR & DMA_SxCR_EN)
178 {
179
180 } // Wait for DMA to finish
181 DMA2->LIFCR |= DMA_LIFCR_CTCIF1; // Clear transfer complete interrupt fl.
182 DMA2_Stream1->CR |= (2UL << DMA_SxCR_CHSEL_Pos); // Select channel 2
183 DMA2_Stream1->CR |= DMA_SxCR_PL_0; // Priority high
184 DMA2_Stream1->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
185 DMA2_Stream1->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
186 DMA2_Stream1->CR |= DMA_SxCR_MINC; // Increment memory address pointer
187 DMA2_Stream1->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
188 // Number of data items to transfer
189 DMA2_Stream1->NDTR = INPUT_COUNT * ADC_NUMS;
190 DMA2_Stream1->PAR = (uint32_t)&ADC3->DR; // Peripheral register address
191 DMA2_Stream1->M0AR = (uint32_t)ADC_samples; // Buffer memory loc. address
192}

198void MEAS_ADC3_scan_start(void)
199{
200 DMA2_Stream1->CR |= DMA_SxCR_EN; // Enable DMA
201 NVIC_ClearPendingIRQ(DMA2_Stream1_IRQn); // Clear pending DMA interrupt
202 NVIC_EnableIRQ(DMA2_Stream1_IRQn); // Enable DMA interrupt in the NVIC
203 ADC3->CR2 |= ADC_CR2_ADON; // Enable ADC3
204 TIM2->CR1 |= TIM_CR1_CEN; // Enable timer
205}

213void DMA2_Stream1_IRQHandler(void)
214{

220 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA
221 while (DMA2_Stream1->CR & DMA_SxCR_EN) // Wait for DMA to finish
222 {
223

225

227 TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
228 ADC3->CR2 &= ~ADC_CR2_ADON; // Disable ADC3
229 ADC3->CR2 &= ~ADC_CR2_DMA; // Disable DMA mode
