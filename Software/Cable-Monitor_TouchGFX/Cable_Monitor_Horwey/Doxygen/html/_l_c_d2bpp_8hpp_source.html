276 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t x)
277 {
278 assert(x < 16);
279 uint16_t xy10 = 16 * x;
280 uint16_t xy00 = 256 - xy10;
281
282 return (c00 * xy00 + c10 * xy10) >> 8;
283 }
284
285 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t c01, uint8_t c11, uint8_t x, uint8_t y)
286 {
287 assert(x < 16 && y < 16);
288 uint16_t xy11 = x * y;
289 uint16_t xy10 = 16 * x - xy11;
290 uint16_t xy01 = 16 * y - xy11;
291 uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
292
293 return (c00 * xy00 + c10 * xy10 + c01 * xy01 + c11 * xy11) >> 8;
294 }
