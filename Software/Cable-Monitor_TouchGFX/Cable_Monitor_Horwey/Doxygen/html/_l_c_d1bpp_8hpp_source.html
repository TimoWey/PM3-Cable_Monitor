192 bwRLEdata(const uint8_t* src = 0)
193 : data(src), thisHalfByte(0), nextHalfByte(0), rleByte(0), firstHalfByte(true), color(0), length(0)

198 void init(const uint8_t* src)
199 {
200 data = src;
201 rleByte = 0;
202 firstHalfByte = true;
203 color = ~0; // Will be flipped to 0 by first call to getNextLength() below
204 if (src)
205 {
206 // Read two half-bytes ahead
207 thisHalfByte = getNextHalfByte();
208 nextHalfByte = getNextHalfByte();
209 getNextLength();
210 }
211 }
212
213 void skipNext(int32_t skip)
214 {
215 for (;;)
216 {
217 if (length > skip) // Is the current length enough?
218 {
219 length -= skip; // Reduce the length
220 skip = 0; // No more to skip
221 break; // Done!
222 }
223 else
224 {
225 skip -= length; // Skip the entire run
226 getNextLength(); // Swap colors and Read length of next run
227 }
228 }
229 }

237 {

241 private:
242 void getNextLength()
243 {
244 length = thisHalfByte; // Length is the next byte
245 // Update read ahead buffer
246 thisHalfByte = nextHalfByte;
247 nextHalfByte = getNextHalfByte();
248 color = ~color; // Update the color of next run
249 // If number after 'length' is 0
250 while (thisHalfByte == 0)
251 {
252 length <<= 4; // Multiply length by 16 and
253 length += nextHalfByte; // add the number after 0
254 // We have used the next two half bytes, read two new ones
255 thisHalfByte = getNextHalfByte();
256 nextHalfByte = getNextHalfByte();
257 }
258 if (length == 0)
259 {
260 getNextLength();
261 }
262 }
263
264 uint8_t getNextHalfByte()
265 {
266 assert(data != 0 && "class bwRLEdata not properly initialized");
267 if (firstHalfByte) // Start of new byte, read data from BW_RLE stream
268 {
269 rleByte = *data++;
270 }
271 const uint8_t len = rleByte & 0xF; // Read lower half
272 rleByte >>= 4; // Shift upper half down to make it ready
273 firstHalfByte = !firstHalfByte; // Toggle 'start of byte'
274 return len;
275 }
276
277 const uint8_t* data; // Pointer to compressed data (BW_RLE)
278 uint8_t thisHalfByte; // The next half byte from the input
279 uint8_t nextHalfByte; // The next half byte after 'thisHalfByte'
280 uint8_t rleByte; // Byte read from compressed data
281 bool firstHalfByte; // Are we about to process first half byte of rleByte?
282 uint8_t color; // Current color
283 int32_t length; // Number of pixels with the given color
284 };
