31void lineFromColor(uint8_t* const ptr, const unsigned count, const uint32_t color, const uint8_t alpha)
32{
33 uint8_t* framebuffer = ptr;
34 const uint8_t* const lineEnd = framebuffer + count * 3;
35 const uint8_t painterRed = Color::getRed(color);
36 const uint8_t painterGreen = Color::getGreen(color);
37 const uint8_t painterBlue = Color::getBlue(color);
38
39 if (alpha == 0xFF)
40 {
41 do
42 {
43 *framebuffer++ = painterBlue;
44 *framebuffer++ = painterGreen;
45 *framebuffer++ = painterRed;
46 } while (framebuffer < lineEnd);
47 }
48 else
49 {
50 do
51 {
52 const uint8_t ialpha = 0xFF - alpha;
53 uint8_t pByte = *framebuffer;
54 *framebuffer++ = LCD::div255(painterBlue * alpha + pByte * ialpha);
55 pByte = *framebuffer;
56 *framebuffer++ = LCD::div255(painterGreen * alpha + pByte * ialpha);
57 pByte = *framebuffer;
58 *framebuffer++ = LCD::div255(painterRed * alpha + pByte * ialpha);
59 } while (framebuffer < lineEnd);
60 }
61}

92void lineFromARGB8888(uint8_t* const ptr, const uint32_t* const data, const unsigned count, const uint8_t alpha)
93{
