179void gyro_disable(void)
180{
181 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
182 /* Disable PC1 and PF8 first */
183 GPIOC->MODER &= ~GPIO_MODER_MODER1_Msk; // Reset mode for PC1
184 GPIOC->MODER |= 1UL << GPIO_MODER_MODER1_Pos; // Set PC1 as output
185 GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
186 HAL_Delay(10); // Wait some time
187 GPIOC->MODER |= 3UL << GPIO_MODER_MODER1_Pos; // Analog PC1 = ADC123_IN11
188 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
189 GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8_Msk; // Reset speed of PF8
190 GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8_Msk; // Reset alternate function of PF8
191 GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8_Msk; // Reset pulup/down of PF8
192 HAL_Delay(10); // Wait some time
193 GPIOF->MODER |= 3UL << GPIO_MODER_MODER8_Pos; // Analog mode PF8 = ADC3_IN4
194}

196// Global Calibration values
197// At distance: 10mm, 50mm and 100mm
198// 0-2: Left Pad
199// 3-5: Right Pad
200// 6-7: Hall Left 1P 1.2A and 5A
201// 8-9: Hall Right 1P 1.2A and 5A
202// 10-11: Hall Left 3P 1.2A and 5A
203// 12-13: Hall Right 3P 1.2A and 5A
204uint32_t Calibration_Data[14]; // The Calibration values are read after init

206// read all 12 flash values since they get erased hen writing a new value
207// proceed to write the 23 values including the new one
208void setCalibrationFlashValues(uint32_t StartSectorAddress, uint32_t value, uint32_t index, uint16_t numberofwords)
209{
210 uint32_t Rx_Data[numberofwords];
211 uint32_t *data_addr = Rx_Data;
212
213 Flash_Read_Data(StartSectorAddress, Rx_Data, numberofwords);
214
215 // Replace the data to write
216 Rx_Data[index] = value;
217
218 Flash_Write_Data(StartSectorAddress, data_addr, numberofwords);
219
220 // store new data in global variable
221 Calibration_Data[index] = value;
222}

939static void BSP_SDRAM_Initialization_Sequence(SDRAM_HandleTypeDef *hsdram, FMC_SDRAM_CommandTypeDef *Command)
940{
941 __IO uint32_t tmpmrd =0;
942
943 /* Step 1: Configure a clock configuration enable command */
944 Command->CommandMode = FMC_SDRAM_CMD_CLK_ENABLE;
945 Command->CommandTarget = FMC_SDRAM_CMD_TARGET_BANK2;
946 Command->AutoRefreshNumber = 1;
947 Command->ModeRegisterDefinition = 0;
948
949 /* Send the command */
950 HAL_SDRAM_SendCommand(hsdram, Command, SDRAM_TIMEOUT);
951
952 /* Step 2: Insert 100 us minimum delay */
953 /* Inserted delay is equal to 1 ms due to systick time base unit (ms) */
954 HAL_Delay(1);
955
956 /* Step 3: Configure a PALL (precharge all) command */
957 Command->CommandMode = FMC_SDRAM_CMD_PALL;
958 Command->CommandTarget = FMC_SDRAM_CMD_TARGET_BANK2;
959 Command->AutoRefreshNumber = 1;
960 Command->ModeRegisterDefinition = 0;
961
962 /* Send the command */
963 HAL_SDRAM_SendCommand(hsdram, Command, SDRAM_TIMEOUT);
964
965 /* Step 4: Configure an Auto Refresh command */
966 Command->CommandMode = FMC_SDRAM_CMD_AUTOREFRESH_MODE;
967 Command->CommandTarget = FMC_SDRAM_CMD_TARGET_BANK2;
968 Command->AutoRefreshNumber = 4;
969 Command->ModeRegisterDefinition = 0;
970
971 /* Send the command */
972 HAL_SDRAM_SendCommand(hsdram, Command, SDRAM_TIMEOUT);
973
974 /* Step 5: Program the external memory mode register */
975 tmpmrd = (uint32_t)SDRAM_MODEREG_BURST_LENGTH_1 |
976 SDRAM_MODEREG_BURST_TYPE_SEQUENTIAL |
977 SDRAM_MODEREG_CAS_LATENCY_3 |
978 SDRAM_MODEREG_OPERATING_MODE_STANDARD |
979 SDRAM_MODEREG_WRITEBURST_MODE_SINGLE;
980
981 Command->CommandMode = FMC_SDRAM_CMD_LOAD_MODE;
982 Command->CommandTarget = FMC_SDRAM_CMD_TARGET_BANK2;
983 Command->AutoRefreshNumber = 1;
984 Command->ModeRegisterDefinition = tmpmrd;
985
986 /* Send the command */
987 HAL_SDRAM_SendCommand(hsdram, Command, SDRAM_TIMEOUT);

989 /* Step 6: Set the refresh rate counter */
990 /* Set the device refresh rate */
