598 static Q15 sine(Q5 i)
599 {
600 Q5 _360 = toQ5<int>(360);
601 i %= _360;
602 if (i < 0)
603 {
604 i += _360;
605 }
606 int16_t fraction = i % Rasterizer::POLY_BASE_SIZE;
607 Q15 sineLow = sine(i.to<int>());
608 if (fraction == 0)
609 {
610 return sineLow;
611 }
612 Q15 sineHigh = sine(i.to<int>() + 1);
613 int32_t remainder;
614 return Q15(muldiv((int)(sineHigh - sineLow), fraction, Rasterizer::POLY_BASE_SIZE, remainder)) + sineLow;
615 }

659 0, 0, 1, 1, 2, 2, 3, 3, 4, 4, 4, 5, 5, 6, 6,
660 7, 7, 8, 8, 9, 9, 9, 10, 10, 11, 11, 12, 12, 13, 13,
661 14, 14, 14, 15, 15, 16, 16, 17, 17, 18, 18, 19, 19, 20, 20,
662 21, 21, 22, 22, 23, 23, 23, 24, 24, 25, 25, 26, 26, 27, 27,
663 28, 28, 29, 29, 30, 31, 31, 32, 32, 33, 33, 34, 34, 35, 35,
664 36, 36, 37, 38, 38, 39, 39, 40, 40, 41, 42, 42, 43, 43, 44, 45
665 };
666 if (q10 < 0)
667 {
668 return -arcsine(-q10);
669 }
670 if (q10 > toQ10<int>(1))
671 {
672 return 0; // error
673 }
674 // sqrt(1/2) as Q10 is "724" so if q10>724 calculate 90-arcsine(sqrt(1-q10^2))
675 if ((int)q10 > 724)

677 return 90 - arcsine(Q10(isqrt((1 << (Rasterizer::POLY_BASE_SHIFT * 4)) - (int)q10 * (int)q10)));
678 }
679 int q7 = ((int)q10 + 3) >> 3; // Round Q10 to nearest Q7
680 return arcsineTable[q7];
681 }

788 static Q5 length(Q5 x, Q5 y)
789 {
790 // Check if x*x or y*y is likely to overflow
791 const int absx = abs((int)x);
792 const int absy = abs((int)y);
793 // Find a scaling factor to make the calulation overflow-safe
794 const int factor = MAX(absx >> 14, absy >> 14);
795 if (factor > 1)
796 {
797 // Recursive call on a smaller number
798 return length(x / factor, y / factor) * factor;
799 }
800 return sqrtQ10(x * x + y * y);
801 }

882 static int _angle(Q5 x, Q5 y, Q5& d)
883 {
884 assert(x >= 0 && y >= 0);
885 // The sine table from 0-45 is more precise, so swap x/y to get the angle in that range
886 if (x < y)
887 {
888 return 90 - _angle(y, x, d);
889 }
890
891 Q5 _1 = toQ5<int>(1); // Used to convert Q5->Q10->Q15
892 d = length(x, y);
893 if (d == 0)
894 {
895 return 0; // Error
896 }
897 int32_t remainder;
898 Q10 dy = Q10(muldiv((int)y, (int)(_1 * _1), (int)d, remainder));
899 return arcsine(dy);
900 }

902 static uint32_t isqrt(uint32_t n)
903 {
904 uint32_t root = 0, bit, trial;
905 bit = (n >= 0x10000) ? 1 << 30 : 1 << 14;
906 do
907 {
908 trial = root + bit;
909 if (n >= trial)
910 {
911 n -= trial;
912 root = trial + bit;
913 }
914 root >>= 1;
915 bit >>= 2;
916 } while (bit);
917 return root;
918 }
