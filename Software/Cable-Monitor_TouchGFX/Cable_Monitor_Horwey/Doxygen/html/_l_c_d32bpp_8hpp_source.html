467 {
468 uint8_t r = (rgb565 & 0xF800) >> 8;
469 r |= r >> 5;
470 uint8_t g = (rgb565 & 0x07E0) >> 3;

490 {
491 if (alphaFg == 255)
492 {
493 destBits[0] = blueFg;
494 destBits[1] = greenFg;
495 destBits[2] = redFg;
496 destBits[3] = 255;

559 FORCE_INLINE_FUNCTION static uint16_t bilinearInterpolate565(uint16_t c00, uint16_t c10, uint16_t c01, uint16_t c11, uint8_t x, uint8_t y)
560 {
561 assert(x < 16 && y < 16);
562 uint32_t a00 = expandRgb565(c00);
563 uint32_t a10 = expandRgb565(c10);
564 uint32_t a01 = expandRgb565(c01);
565 uint32_t a11 = expandRgb565(c11);
566
567 uint8_t xy = (x * y) >> 3;
568 return compactRgb565((a00 * (32 - 2 * y - 2 * x + xy) + a10 * (2 * x - xy) + a01 * (2 * y - xy) + a11 * xy) >> 5);
569 }

589 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t c01, uint8_t c11, uint8_t x, uint8_t y)
590 {
591 assert(x < 16 && y < 16);
592 uint16_t xy11 = x * y;
593 uint16_t xy10 = 16 * x - xy11;
594 uint16_t xy01 = 16 * y - xy11;

596
