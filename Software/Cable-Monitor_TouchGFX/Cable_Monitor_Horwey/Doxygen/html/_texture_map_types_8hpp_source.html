176FORCE_INLINE_FUNCTION fixed28_4 fixed28_4Mul(fixed28_4 a, fixed28_4 b)
177{
178 if (a == 0 || b == 0)
179 {
180 return 0;
181 }
182 if ((a * b) / b == a)
183 {
184 return (a * b) / 16;
185 }
186
187 // Rewrite "b = max_b * num_max_b + rem_b" so that a * max_b does not overflow.
188 int sign = 1;
189 if (b < 0)
190 {
191 sign = -sign;
192 b = -b;
193 }
194 if (a < 0)
195 {
196 sign = -sign;
197 a = -a;
198 }
199 if (a < b)
200 {
201 fixed28_4 tmp = a;
202 a = b;
203 b = tmp;
204 }
205 int32_t max_b = 0x7FFFFFFF / a; // Max b value that can be multiplied with a without overflow
206 int32_t num_max_b = b / max_b; // How many times do we have to multiply with "max_b" to get to "b"
207 int32_t rem_b = b - max_b * num_max_b; // plus some remainder.
208 int32_t max_prod = a * max_b;
209 int32_t result = sign * (num_max_b * (max_prod / 16) + (num_max_b * (max_prod % 16) + rem_b * a) / 16);
210 return result;
211}

220FORCE_INLINE_FUNCTION int32_t ceil28_4(fixed28_4 value)
221{

245FORCE_INLINE_FUNCTION void floorDivMod(int32_t numerator, int32_t denominator, int32_t& floor, int32_t& mod)
246{
247 assert(denominator > 0); // We assume it's positive
248 if (numerator >= 0)
249 {
250 // Positive case, C is okay
251 floor = numerator / denominator;
252 mod = numerator % denominator;
253 }
254 else
255 {
256 // Numerator is negative, do the right thing
257 floor = -((-numerator) / denominator);
258 mod = (-numerator) % denominator;
259 if (mod)
260 {
261 // There is a remainder
262 floor--;
263 mod = denominator - mod;
264 }
