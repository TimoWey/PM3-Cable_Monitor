167 unsigned calculateAlpha(int area) const
168 {
169 int cover = area >> (POLY_BASE_SHIFT * 2 + 1 - AA_SHIFT);
170
171 if (cover < 0)
172 {
173 cover = -cover;
174 }
175 if (fillingRule == FILL_EVEN_ODD)
176 {
177 cover &= AA_2MASK;
178 if (cover > AA_NUM)
179 {
180 cover = AA_2NUM - cover;
181 }
182 }
183 if (cover > AA_MASK)
184 {
185 cover = AA_MASK;
186 }
187 return cover;
188 }

196 bool render(const AbstractPainter* const painter, uint8_t* buf, int16_t stride, uint8_t xAdjust, uint8_t global_alpha)
197 {
198 const Cell* curCell = outline.closeOutlineAndSortCells(); // has side effects, so it must be first in this function
199 unsigned numCells = outline.getNumCells();
200 if (numCells == 0)
201 {
202 return true;
203 }
204
205 // Not enough memory allocated, the outline will look wrong. Do not proceed with the drawing.
206 if (outline.wasOutlineTooComplex())
207 {
208 return false;
209 }
210
211 int cover = 0;
212 int old_y = curCell->y;
213 int widget_y = old_y + offsetY;
214 uint8_t* buf_ptr = buf + old_y * stride;
215 numCells--;
216 for (;;)
217 {
218 const Cell* startCell = curCell;
219
220 const int start_x = curCell->x;
221 int x = start_x;
222 const int y = curCell->y;
223 if (y != old_y)
224 {
225 old_y = y;
226 widget_y = old_y + offsetY;
227 buf_ptr = buf + old_y * stride;
228 }
229
230 int area = startCell->area;
231 cover += startCell->cover;
232
233 // Accumulate all start cells
234 while (numCells-- > 0)
235 {
236 curCell++;
237 if (curCell->x != start_x || curCell->y != y)
238 {
239 break;
240 }
241 area += curCell->area;
242 cover += curCell->cover;
243 }
244
245 if (area)
246 {
247 if (x >= 0 && x < width)
248 {
249 const int8_t alpha = LCD::div255(calculateAlpha((cover << (POLY_BASE_SHIFT + 1)) - area) * global_alpha);
250 if (alpha)
251 {
252 painter->paint(buf_ptr, x + xAdjust, x + offsetX, widget_y, 1, alpha);
253 }
254 }
255 x++;
256 }
257
258 if (numCells == unsigned(-1))
259 {
260 break;
261 }
262
263 int count = curCell->x - x;
264 if (count > 0)
265 {
266 if (x < 0)
267 {
268 count += x;
269 x = 0;
270 }
271 if (count > 0)
272 {
273 if (x + count >= width)
274 {
275 count = width - x;
276 }
277 if (count > 0)
278 {
279 const int8_t alpha = LCD::div255(calculateAlpha(cover << (POLY_BASE_SHIFT + 1)) * global_alpha);
280 if (alpha)
281 {
282 painter->paint(buf_ptr, x + xAdjust, x + offsetX, widget_y, count, alpha);
283 }
284 }
285 }
286 }
287 }
288 return true;
289 }
