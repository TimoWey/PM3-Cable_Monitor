177 void nextMoveAnimationStep()
178 {
179 if (moveAnimationRunning)
180 {
181 moveAnimationCounter++;
182 if (moveAnimationCounter >= moveAnimationDelay)
183 {
184 // Adjust the used animationCounter for the startup delay
185 uint32_t actualAnimationCounter = moveAnimationCounter - moveAnimationDelay;
186
187 int16_t deltaX = moveAnimationXEquation(actualAnimationCounter, 0, moveAnimationEndX - moveAnimationStartX, moveAnimationDuration);
188 int16_t deltaY = moveAnimationYEquation(actualAnimationCounter, 0, moveAnimationEndY - moveAnimationStartY, moveAnimationDuration);
189
190 T::moveTo(moveAnimationStartX + deltaX, moveAnimationStartY + deltaY);
191
192 if (moveAnimationCounter >= (uint32_t)(moveAnimationDelay + moveAnimationDuration))
193 {
194 // End of animation

196 moveAnimationCounter = 0;
