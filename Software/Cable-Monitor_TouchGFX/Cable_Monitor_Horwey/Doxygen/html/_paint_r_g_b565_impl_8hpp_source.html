50FORCE_INLINE_FUNCTION uint16_t alphaBlend(uint16_t R, uint16_t G, uint16_t B, uint16_t bufpix, uint8_t alpha)
51{
52 const uint8_t ialpha = 0xFF - alpha;
53 return (((R * alpha + (bufpix & RMASK) * ialpha) / 255) & RMASK) |
54 (((G * alpha + (bufpix & GMASK) * ialpha) / 255) & GMASK) |
55 (((B * alpha + (bufpix & BMASK) * ialpha) / 255) & BMASK);
56}

127void lineFromARGB8888(uint16_t* const ptr, const uint32_t* const data, const unsigned count, const uint8_t alpha)
128{
129 uint16_t* framebuffer = ptr;
130 const uint32_t* bitmapPointer = data;
131 const uint16_t* const chunkend = framebuffer + count;
132 do
133 {
134 const uint8_t srcAlpha = (*bitmapPointer) >> 24;
135 const uint8_t a = LCD::div255(alpha * srcAlpha);
136 if (a == 0xFF)
137 {
138 *framebuffer = getNativeColor(*bitmapPointer);
139 }
140 else if (a)
141 {
142 const uint32_t newpix = *bitmapPointer;
143 *framebuffer = alphaBlend((newpix >> 8) & RMASK, (newpix >> 5) & GMASK, (newpix >> 3) & BMASK, *framebuffer, a);
144 }
145 bitmapPointer++;
146 } while (++framebuffer < chunkend);
147}

149void lineFromL8RGB888(uint16_t* const ptr, const uint8_t* const data, const unsigned count, const uint8_t alpha)
150{
151 uint16_t* framebuffer = ptr;
152 const uint8_t* bitmapPointer = data;
153 const uint16_t* const chunkend = framebuffer + count;
154 if (alpha == 0xFF)
155 {
156 do
157 {
158 const uint8_t* src = &blendL8CLUT[*bitmapPointer++ * 3];
159 // Use alpha from covers directly
160 const uint8_t blue = *src++;
161 const uint8_t green = *src++;
162 const uint8_t red = *src;
163 *framebuffer = ((red << 8) & RMASK) | ((green << 3) & GMASK) | ((blue >> 3) & BMASK);
164 } while (++framebuffer < chunkend);
165 }
166 else
167 {
168 do
169 {
170 const uint8_t* src = &blendL8CLUT[*bitmapPointer++ * 3];
171 // Use alpha from covers directly
172 const uint8_t blue = *src++;
173 const uint8_t green = *src++;
174 const uint8_t red = *src;
175 const uint8_t ialpha = 0xFF - alpha;
176 const uint16_t bufpix = *framebuffer;
177 uint8_t fbr = Color::getRedFromRGB565(bufpix);
178 uint8_t fbg = Color::getGreenFromRGB565(bufpix);
179 uint8_t fbb = Color::getBlueFromRGB565(bufpix);
180 *framebuffer = ((LCD::div255(red * alpha + fbr * ialpha) << 8) & RMASK) | ((LCD::div255(green * alpha + fbg * ialpha) << 3) & GMASK) | ((LCD::div255(blue * alpha + fbb * ialpha) >> 3) & BMASK);
181 } while (++framebuffer < chunkend);
182 }
