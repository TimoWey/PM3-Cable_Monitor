36template <Direction templateDirection>
37class WipeTransition : public Transition
38{
39public:
45 WipeTransition(const uint8_t transitionSteps = 20)
46 : Transition(),
47 animationSteps(transitionSteps),
48 animationCounter(0),
49 calculatedValue(0),
50 solid()
51 {
52 switch (templateDirection)
53 {
54 case EAST:
55 case WEST:
56 targetValue = HAL::DISPLAY_WIDTH;
57 break;
58 case NORTH:
59 case SOUTH:
60 targetValue = HAL::DISPLAY_HEIGHT;
61 break;
62 }
63
64 // Ensure that the solid area covers the entire screen
65 solid.setPosition(0, 0, HAL::DISPLAY_WIDTH, HAL::DISPLAY_HEIGHT);
66 }
67
72 virtual void handleTickEvent()
73 {
74 Transition::handleTickEvent();
75 animationCounter++;
76
77 // Calculate new position or stop animation
78 if (animationCounter > animationSteps)
79 {
80 // Final step: stop the animation
81 done = true;
82 animationCounter = 0;
83 return;
84 }
85
86 // Calculate value in [0;targetValue]
87 calculatedValue = EasingEquations::cubicEaseOut(animationCounter, 0, targetValue, animationSteps);
88
89 // Note: Result of "calculatedValue & 1" is compiler dependent for negative values of calculatedValue
90 if ((calculatedValue % 2) != 0)
91 {
92 // Optimization: calculatedValue is odd, add 1/-1 to move drawables modulo 32 bits in framebuffer
93 calculatedValue += (calculatedValue > 0 ? 1 : -1);
94 }
95
96 // calculatedValue is the width/height of the visible area
97
98 switch (templateDirection)
99 {
100 case EAST:
101 {
102 // Cover must have width of remaining part
103 const uint16_t prevSolidWidth = solid.getWidth();
104 solid.setWidth(HAL::DISPLAY_WIDTH - calculatedValue);
105
106 // Invalidate the uncovered part
107 const uint16_t delta = prevSolidWidth - solid.getWidth();
108 Rect r(solid.getWidth(), 0, delta, HAL::DISPLAY_HEIGHT);
109 screenContainer->invalidateRect(r);
110 break;
111 }
112 case WEST:
113 {
114 // Cover must have width of remaining part and start after uncovered
115 const uint16_t prevSolidPos = solid.getX();
116 solid.setWidth(HAL::DISPLAY_WIDTH - calculatedValue);
117 solid.setX(calculatedValue);
118
119 // Invalidate the uncovered part
120 const uint16_t delta = calculatedValue - prevSolidPos;
121 Rect r(prevSolidPos, 0, delta, HAL::DISPLAY_HEIGHT);
122 screenContainer->invalidateRect(r);
123 break;
124 }
125 case NORTH:
126 {
127 // Cover must have height of remaining part and start after uncovered
128 const uint16_t prevSolidPos = solid.getY();
129 solid.setHeight(HAL::DISPLAY_HEIGHT - calculatedValue);
130 solid.setY(calculatedValue);
131
132 // Invalidate the uncovered part
133 const uint16_t delta = calculatedValue - prevSolidPos;
134 Rect r(0, prevSolidPos, HAL::DISPLAY_WIDTH, delta);
135 screenContainer->invalidateRect(r);
136 break;
137 }
138 case SOUTH:
139 {
140 // Cover must have height of remaining part
141 const uint16_t prevSolidHeight = solid.getHeight();
142 solid.setHeight(HAL::DISPLAY_HEIGHT - calculatedValue);
143
144 // Invalidate the uncovered part
145 const uint16_t delta = prevSolidHeight - solid.getHeight();
146 Rect r(0, solid.getHeight(), HAL::DISPLAY_WIDTH, delta);
147 screenContainer->invalidateRect(r);
148 break;
149 }
150 }
151
152 // The WipeTransition only draws to parts of the non-TFT
153 // framebuffer. To avoid glitches in Double buffering mode
154 // both framebuffers must be made identical.
155 if (animationCounter == 1 && HAL::USE_DOUBLE_BUFFERING)
156 {
157 // Synchronize framebuffers
158 Application::getInstance()->copyInvalidatedAreasFromTFTToClientBuffer();
159 }
160 }
161
162 virtual void tearDown()
163 {
164 screenContainer->remove(solid);
165 }
166
167 virtual void init()
168 {
169 Transition::init();
170 // Add the solid (and not-drawing-anything) widget on top to cover the other widgets
171 screenContainer->add(solid);
172 }
173
178 virtual void invalidate()
179 {
180 }
181
182private:
183 const uint8_t animationSteps;
184 uint8_t animationCounter;
185 int16_t targetValue;
186 int16_t calculatedValue;
187 FullSolidRect solid;
188};
