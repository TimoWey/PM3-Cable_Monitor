137FFT calculate_freq_and_signalstrength(float32_t input_samples[])
138{
139 // remove DC component
140 for (uint32_t i = 0; i < SAMPLE_LEN; i++) {
141 input_samples[i] -= 2047.5;
142 }
143 // Create an array for the FFT output
144 float32_t fft_complex_output[2 * SAMPLE_LEN];
145 // Create an instance of the FFT structure
146 arm_rfft_fast_instance_f32 fft_struct;
147 // Initialize the FFT structure
148 arm_rfft_fast_init_f32(&fft_struct, SAMPLE_LEN);
149 // Perform the FFT on the input samples
150 arm_rfft_fast_f32(&fft_struct, input_samples, fft_complex_output, 0);
151 // Calculate magnitude of complex numbers
152 arm_cmplx_mag_f32(fft_complex_output, fft_complex_output, SAMPLE_LEN / 2);
153 // Find the index of the maximum magnitude in the first half of the array
154 uint32_t maxIndex;
155 float32_t maxValue;
156 arm_max_f32(fft_complex_output, SAMPLE_LEN / 2, &maxValue, &maxIndex);
157 // Create an instance of the FFT structure
158 FFT fft;
159 // Calculate the main frequency
160 float main_frequency =
161 (float)maxIndex * ((float)MEAS_get_samp_freq()) / SAMPLE_LEN;
162 uint8_t given_frequency = 0;
163 // Check if the main frequency is in the range of 45-55 Hz
164 if (main_frequency <= 55 && main_frequency >= 45) {
165 // If yes, round the given frequency to 50 Hz
166 given_frequency = 50;
167 } else if (main_frequency <= 65 && main_frequency >= 55) {
168 // If yes, round the given frequency to 60 Hz
169 given_frequency = 60;
170 } else given_frequency = 0;
171 // calculate the index of the given frequency
172 uint32_t index = (uint32_t)given_frequency * SAMPLE_LEN / MEAS_get_samp_freq();
173 // set the signal strength to 0
174 float signal_strength = 0;
175 // check if the given frequency is 50 or 60 Hz
176 if (given_frequency != 0) {
177 // calculate the signal strength
178 signal_strength = 2 * fft_complex_output[index] / SAMPLE_LEN;
179 fft.error = CALC_ERROR_NONE;
180 } else fft.error = CALC_ERROR_FREQUENCY; // if the given frequency is not 50 or 60 Hz
181 // return the calculated values
182 fft.main_freq = main_frequency;
183 fft.signal_strength = signal_strength;
184 return fft;
185}

265SINGLE_MEAS single_measurement(uint8_t Phase) {
266 // Start the measurement
267 uint32_t* samples = MEAS_start_measure();
268 // Create an instance of the SINGLE_MEAS structure
269 SINGLE_MEAS single_meas;
270 // Create an instance of the FFT structure
271 FFT fft;
272
273 // Set the error to none
274 single_meas.error = CALC_ERROR_NONE;
275
276 static float32_t samples_PR[SAMPLE_LEN];
277 static float32_t samples_PL[SAMPLE_LEN];
278 static float32_t samples_HSR[SAMPLE_LEN];
279 static float32_t samples_HSL[SAMPLE_LEN];
280
281 float32_t current_l = 0;
282 float32_t current_r = 0;
283
284 // Convert the samples to the specific channel
285 for (uint32_t i = 0; i < 64; i++) {
286 samples_PR[i] = (float32_t)samples[(i * CHANNEL_NUM)];
287 samples_PL[i] = (float32_t)samples[(i * CHANNEL_NUM) + 1];
288 samples_HSR[i] = (float32_t)samples[(i * CHANNEL_NUM) + 2];
289 samples_HSL[i] = (float32_t)samples[(i * CHANNEL_NUM) + 3];
290 }
291 // Calculate the Signal Strength and the main frequency for each channel using FFT
292 fft = calculate_freq_and_signalstrength(samples_PR);
293 float32_t signal_strength_PR = fft.signal_strength;
294 float32_t main_frequency_PR = fft.main_freq;
295 fft = calculate_freq_and_signalstrength(samples_PL);
296 float32_t signal_strength_PL = fft.signal_strength;
297 float32_t main_frequency_PL = fft.main_freq;
298 fft = calculate_freq_and_signalstrength(samples_HSR);
299 float32_t signal_strength_HSR = fft.signal_strength;
300 fft = calculate_freq_and_signalstrength(samples_HSL);
301 float32_t signal_strength_HSL = fft.signal_strength;
302 // Calculate the main frequency
303 single_meas.frequency = (main_frequency_PR + main_frequency_PL) / 2;
304 // Calculate distance and angle
305 DISTANCE_ANGLE dist_angle = calculate_distance_and_angle(signal_strength_PR, signal_strength_PL);
306 single_meas.distance = dist_angle.distance;
307 single_meas.angle = dist_angle.angle;
308
309 // Calculate the current using the magnetic coefficients for 5A
310 if(Phase == SINGLE_PHASE){
311 current_l = a_magn2_l_1P * single_meas.distance * signal_strength_HSL;
312 current_r = a_magn2_r_1P * single_meas.distance * signal_strength_HSR;
313 single_meas.current = (current_l + current_r) / 2;
314 } else if(Phase == THREE_PHASE){
315 if (signal_strength_HSL > signal_strength_HSR){
316 single_meas.current = a_magn2_l_3P * single_meas.distance * signal_strength_HSL;
317 } else {
318 single_meas.current = a_magn2_r_3P * single_meas.distance * signal_strength_HSR;
319 }
320 }
321
322 // Check if the current is smaller than 4 A -> use the magnetic coefficients for 1.2 A
323 if (single_meas.current < 4){
324 if(Phase == SINGLE_PHASE){
325 current_l = a_magn1_l_1P * single_meas.distance * signal_strength_HSL;
326 current_r = a_magn1_r_1P * single_meas.distance * signal_strength_HSR;
327 single_meas.current = (current_l + current_r) / 2;
328 } else if(Phase == THREE_PHASE){
329 if (signal_strength_HSL > signal_strength_HSR){
330 single_meas.current = a_magn1_l_3P * single_meas.distance * signal_strength_HSL;
331 } else {
332 single_meas.current = a_magn1_r_3P * single_meas.distance * signal_strength_HSR;
333 }
334 }
335 }
336
337 // ERROR TESTING:
338
339 // Check if distance is close enough for current calculation and if the current is out of the range of 0-10 A
340 if (single_meas.distance > 30 || (single_meas.current <= 0 && single_meas.current >= 10)){
341 single_meas.error = CALC_ERROR_TOO_FAR_AWAY;
342 }
343
344 // Check if the current is larger than 10 A -> CALL ERROR OVERCURRENT
345 if (single_meas.current > 10){
346 single_meas.error = CALC_ERROR_OVERCURRENT;
347 }
348
349 // Check if the distance is larger than 300 mm or smaller than 0 -> CALL ERROR DISCONNECT
350 if (single_meas.distance > 250){
351 single_meas.error = CALC_ERROR_DISCONNECT;
352 single_meas.angle = 0;
353 }
354
355 // Check if the frequency is 0 -> CALL ERROR FREQUENCY
356 if (single_meas.frequency == 0){
357 single_meas.error = CALC_ERROR_FREQUENCY;
358 }
359
360 // OPTIMIZE:
361
362 // Limit the angle to +/- 90 degrees
363 if (single_meas.angle > 90){
364 single_meas.angle = 90;
365 }else if (single_meas.angle < -90){
366 single_meas.angle = -90;
367 }
368
369 // BUZZER LED
370 if(single_meas.distance > 0 && single_meas.distance < 200 && (State_BUZZER == false))
371 TIM14->CCR1 = 50 + (single_meas.distance/4);
372 else
373 TIM14->CCR1 = 0;
374
375 if(single_meas.distance > 0 && single_meas.distance < 200 && (State_LED == false))
376 TIM5->CCR1 = abs(100 - (int)(single_meas.distance/2));
377 else
378 TIM5->CCR1 = 0;
379 // Return the calculated values
380 return single_meas;
381}

394ACCU_MEAS accurate_measurement(uint8_t Phase){
395 // Create an instance of the ACCU_MEAS structure
396 ACCU_MEAS accu_meas;
397
398 // set the error to none
399 accu_meas.error_accu = CALC_ERROR_NONE;
400
401 static float32_t distance[ACCURATE_MEASUREMENT_LOOPS];

403 static float32_t frequency[ACCURATE_MEASUREMENT_LOOPS];

412
413 // Check if there is an error in the single measurement
414 if (single_meas.error != CALC_ERROR_NONE){
415 accu_meas.error_single = single_meas.error;
416 }
417 }
418 // Calculate the mean value of the distance
419 arm_mean_f32(distance, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.distance);
420 // Calculate standard deviation of the distance
421 arm_std_f32(distance, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.distance_std_dev);
422 // Calculate the mean value of the angle
423 arm_mean_f32(angle, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.angle);
424 // Calculate standard deviation of the angle
425 arm_std_f32(angle, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.angle_std_dev);
426 // Calculate the mean value of the frequency
427 arm_mean_f32(frequency, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.frequency);
428 // Calculate standard deviation of the frequency
429 arm_std_f32(frequency, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.frequency_std_dev);
430 // Calculate the mean value of the current
431 arm_mean_f32(current, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.current);
432 // Calculate standard deviation of the current
433 arm_std_f32(current, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.current_std_dev);
434
435 // Check if the standard deviation is larger than 100
436 if (accu_meas.distance_std_dev > 100 || accu_meas.angle_std_dev > 100 || accu_meas.frequency_std_dev > 100 || accu_meas.current_std_dev > 100){
437 accu_meas.error_accu = CALC_ERROR_DEVIATION_TOO_HIGH;
438 } else accu_meas.error_accu = CALC_ERROR_NONE;
439
440 // Return the calculated values
441 return accu_meas;
442}

455DISTANCE_ANGLE calculate_distance_and_angle(float32_t signal_strength_r, float32_t signal_strength_l){
456
457 // Create an instance of the DISTANCE_ANGLE structure
458 DISTANCE_ANGLE dist_angle;
459
460 // Calculate the distance using the coefficients and the actual signal strength
461 float32_t calc_distance = a_r + (b_r / signal_strength_r) + (c_r / POW2(signal_strength_r));
462 // Set the calculated values
463 dist_angle.distance_r = calc_distance;
464
465 // Calculate the distance using the coefficients and the actual signal strength
466 calc_distance = a_l + (b_l / signal_strength_l) + (c_l / POW2(signal_strength_l));
467 // Set the calculated values
468 dist_angle.distance_l = calc_distance;
469
470 // Calculate the distance
471 dist_angle.distance = (dist_angle.distance_l + dist_angle.distance_r) / 2;
472
473 // Calculate the angle using the trigonometry
474 dist_angle.angle = atanf((dist_angle.distance_l - dist_angle.distance_r) / (2 * D_P)) * (180 / PI);
475
476 // Return the calculated values
477 return dist_angle;
478}

531void calculate_coefficients_single_pad(float32_t s[], float32_t d[], float32_t* a, float32_t* b, float32_t* c) {
532 // s = x_0, x_1, x_2 d= y_0, y_1, y_2
533 // Calculate the coefficients for the distance approximation from an approximation based on a second degree polynomial
534 // a=((x_0^(2)*(x_1-x_2)*y_0-x_0*(x_1^(2)*y_1-x_2^(2)*y_2)+x_1*(x_1*y_1-x_2*y_2)*x_2)/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
535 // -> Result of Nspire CAS
536
537 *a = (POW2(s[0]) * (s[1] - s[2]) * d[0] - s[0] * (POW2(s[1]) * d[1] - POW2(s[2]) * d[2]) + s[1] * (s[1] * d[1] - s[2] * d[2]) * s[2])
538 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
539 // b = ((−(x_0^(2)*(x_1^(2)*(y_0-y_1)-x_2^(2)*(y_0-y_2))+x_1^(2)*x_2^(2)*(y_1-y_2)))/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
540 // -> Result of Nspire CAS
541
542 *b = (-(POW2(s[0]) * (POW2(s[1]) * (d[0] - d[1]) - POW2(s[2]) * (d[0] - d[2])) + POW2(s[1]) * POW2(s[2]) * (d[1] - d[2])))
543 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
544
545 // c=((x_0*(x_0*(x_1*(y_0-y_1)-x_2*(y_0-y_2))+x_1*x_2*(y_1-y_2))*x_1*x_2)/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
546 // -> Result of Nspire CAS
547 *c = ((s[0] * (s[0] * (s[1] * (d[0] - d[1]) - s[2] * (d[0] - d[2])) + s[1] * s[2] * (d[1] - d[2])) * s[1] * s[2]))
548 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
549}
