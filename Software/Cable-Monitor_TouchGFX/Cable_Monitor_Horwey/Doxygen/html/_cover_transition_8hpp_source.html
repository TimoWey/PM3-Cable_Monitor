44 CoverTransition(const uint8_t transitionSteps = 20)
45 : Transition(),
46 animationSteps(transitionSteps),
47 animationCounter(0),
48 calculatedValue(0),

52 {
53 case EAST:
54 targetValue = -HAL::DISPLAY_WIDTH;
55 break;
56 case WEST:
57 targetValue = HAL::DISPLAY_WIDTH;
58 break;
59 case NORTH:
60 targetValue = HAL::DISPLAY_HEIGHT;
61 break;
62 case SOUTH:
63 targetValue = -HAL::DISPLAY_HEIGHT;
64 break;
65 default:
66 done = true;
67 // Nothing to do here
68 break;
69 }
70
71 // Ensure that the solid area covers the entire screen
72 solid.setPosition(0, 0, HAL::DISPLAY_WIDTH, HAL::DISPLAY_HEIGHT);
73 }

80 virtual void handleTickEvent()
81 {
82 Transition::handleTickEvent();
83
84 // Calculate new position or stop animation
85 animationCounter++;
86 if (animationCounter > animationSteps)
87 {
88 // Final step: stop the animation
89 done = true;
90 animationCounter = 0;
91 return;
92 }
93
94 const int16_t oldValue = calculatedValue;
95
96 // Calculate value in [0;targetValue]
97 calculatedValue = EasingEquations::cubicEaseOut(animationCounter, 0, targetValue, animationSteps);
98
99 // Note: Result of "calculatedValue & 1" is compiler dependent for negative values of calculatedValue
100 if ((calculatedValue % 2) != 0)
101 {
102 // Optimization: calculatedValue is odd, add 1/-1 to move drawables modulo 32 bits in framebuffer
103 calculatedValue += (calculatedValue > 0 ? 1 : -1);
104 }
105
106 // Convert the calculated value to delta value relative to current moved-to position
107 const int16_t relativeValue = calculatedValue - oldValue;
108
109 // The Cover Transition only draws to parts of the non-TFT framebuffer. To avoid glitches
110 // In Double buffering mode both framebuffers must be identical.
111 if (animationCounter == 1 && HAL::USE_DOUBLE_BUFFERING)
112 {
113 // Synchronize framebuffers
114 Application::getInstance()->copyInvalidatedAreasFromTFTToClientBuffer();
115 }
116
117 Rect rect;
118 switch (templateDirection)
119 {
120 case EAST:
121 rect.x = HAL::DISPLAY_WIDTH + calculatedValue;
122 rect.y = 0;
123 rect.width = -calculatedValue;
124 rect.height = HAL::DISPLAY_HEIGHT;
125 break;
126 case WEST:
127 rect.x = 0;
128 rect.y = 0;
129 rect.width = calculatedValue;
130 rect.height = HAL::DISPLAY_HEIGHT;
131 break;
132 case NORTH:
133 rect.x = 0;
134 rect.y = 0;
135 rect.width = HAL::DISPLAY_WIDTH;
136 rect.height = calculatedValue;
137 break;
138 case SOUTH:
139 rect.x = 0;
140 rect.y = HAL::DISPLAY_HEIGHT + calculatedValue;
141 rect.width = HAL::DISPLAY_WIDTH;
142 rect.height = -calculatedValue;
143 break;
144 }
145 Application::getInstance()->invalidateArea(rect);
146
147 // Move children with delta value for X or Y
148 Drawable* d = screenContainer->getFirstChild();
149 while (d)
150 {
151 switch (templateDirection)
152 {
153 case EAST:
154 case WEST:
155 d->setX(d->getX() + relativeValue);
156 break;
157 case NORTH:
158 case SOUTH:
159 d->setY(d->getY() + relativeValue);
160 break;
161 }
162
163 d = d->getNextSibling();
164 }
165 }
166
167 virtual void tearDown()
168 {
169 screenContainer->remove(solid);
170 }
171

173 {
174 Transition::init();
175
176 // Move snapshot and its children with delta value for X or Y
177 Drawable* d = screenContainer->getFirstChild();
178 while (d)

181 {
