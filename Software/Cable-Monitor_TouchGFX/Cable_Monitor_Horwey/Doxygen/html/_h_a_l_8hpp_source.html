958 bool setFrameRefreshStrategy(FrameRefreshStrategy s)
959 {
960 if (s == REFRESH_STRATEGY_DEFAULT || s == REFRESH_STRATEGY_PARTIAL_FRAMEBUFFER)
961 {
962 refreshStrategy = s;
963 return true;
964 }
965 if (s == REFRESH_STRATEGY_OPTIM_SINGLE_BUFFER_TFT_CTRL)
966 {
967 // Perform sanity checks. This strategy requires
968 // - task delay function
969 // - a TFT controller (+ an impl of getTFTCurrentLine())
970 // - single buffering
971 if (taskDelayFunc != 0 && getTFTCurrentLine() != 0xFFFF && !USE_DOUBLE_BUFFERING)
972 {
973 refreshStrategy = s;
974 return true;
975 }
976 return false;
977 }
978 // Unknown strategy
979 return false;
980 }
