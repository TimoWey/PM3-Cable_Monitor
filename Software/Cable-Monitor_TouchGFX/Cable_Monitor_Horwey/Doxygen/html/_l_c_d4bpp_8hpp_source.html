185 FORCE_INLINE_FUNCTION static uint8_t getPixel(const uint8_t* addr, int offset)
186 {
187 uint8_t data = addr[offset / 2];

216 {
217 uint8_t data = addr[offset / 2];
218#ifdef USE_LSB
219 addr[offset / 2] = (offset & 1) ? (data & 0x0F) | (value << 4) : (data & 0xF0) | value;

301 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t x)
302 {
303 assert(x < 16);
304 uint16_t xy10 = 16 * x;
305 uint16_t xy00 = 256 - xy10;
306
307 return (c00 * xy00 + c10 * xy10) >> 8;
308 }
309
310 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t c01, uint8_t c11, uint8_t x, uint8_t y)
311 {
312 assert(x < 16 && y < 16);
313 uint16_t xy11 = x * y;
314 uint16_t xy10 = 16 * x - xy11;
315 uint16_t xy01 = 16 * y - xy11;
316 uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
317
318 return (c00 * xy00 + c10 * xy10 + c01 * xy01 + c11 * xy11) >> 8;
319 }
