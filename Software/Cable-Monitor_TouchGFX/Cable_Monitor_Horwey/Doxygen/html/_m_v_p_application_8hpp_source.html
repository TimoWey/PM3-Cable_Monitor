160template <class ScreenType, class PresenterType, class TransType, class ModelType>
161PresenterType* makeTransition(Screen** currentScreen, Presenter** currentPresenter, MVPHeap& heap, Transition** currentTrans, ModelType* model)
162{
163 assert(sizeof(ScreenType) <= heap.screenStorage.element_size() && "View allocation error: Check that all views are added to FrontendHeap::ViewTypes");
164 assert(sizeof(PresenterType) <= heap.presenterStorage.element_size() && "Presenter allocation error: Check that all presenters are added to FrontendHeap::PresenterTypes");
165 assert(sizeof(TransType) <= heap.transitionStorage.element_size() && "Transition allocation error: Check that all transitions are added to FrontendHeap::TransitionTypes");
166
167 prepareTransition(currentScreen, currentPresenter, currentTrans);
168
169 TransType* newTransition = new (&heap.transitionStorage.at<TransType>(0)) TransType;
170 ScreenType* newScreen = new (&heap.screenStorage.at<ScreenType>(0)) ScreenType;
171 PresenterType* newPresenter = new (&heap.presenterStorage.at<PresenterType>(0)) PresenterType(*newScreen);
172 *currentTrans = newTransition;
173 *currentPresenter = newPresenter;
174 *currentScreen = newScreen;
175 model->bind(newPresenter);
176 newPresenter->bind(model);
177 newScreen->bind(*newPresenter);
178
179 finalizeTransition((Screen*)newScreen, (Presenter*)newPresenter, (Transition*)newTransition);
180
181 return newPresenter;
182}
