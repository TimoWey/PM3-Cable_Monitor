61 void setGradientEndPoints(float startX, float startY, float endX, float endY, float width, float height, const Matrix3x3& m)
62 {
63 const Matrix3x3::Point start = m.affineTransform(startX, startY);
64 const Matrix3x3::Point end = m.affineTransform(endX, endY);
65
66 isVertical = (start.x == end.x);
67 isHorizontal = (start.y == end.y);
68
69 if (!isVertical && !isHorizontal)
70 {
71 // Calculate slope and position of first color line and horizontal distance to last
72 // Note: startX, startY, endX, endY are given in bounding box coordinate system [0;width]x[0;height]
73 // Color lines are perpendicular to gradient vector (cvy, cvx), in the square bounding-box coordinate system [0;1]x[0;1].
74 const float cvx = endX - startX;
75 const float cvy = endY - startY;
76
77 // So we scale gradient back from width x height
78 const float cvxbb = cvx / width;
79 const float cvybb = cvy / height;
80
81 // Equal Colors vector is orthogonal to gradient vector, but scale to real size of shape
82 const float colorlinedx = -cvybb * width;
83 const float colorlinedy = cvxbb * height;
84
85 // Rotate color vector according to transformation
86 const Matrix3x3::Point transformed = m.affineTransform(startX + colorlinedx, startY + colorlinedy);
87
88 // Translate back to startX,Y
89 const float transformedlinedx = transformed.x - start.x;
90
91 // If the color line has dx==0, the gradient (not the color line) is horizontal
92 if (transformedlinedx == 0.0f)
93 {
94 isHorizontal = true;
95 }
96 else
97 {
98 const float transformedlinedy = transformed.y - start.y;
99
100 clSlope = transformedlinedy / transformedlinedx;
101
102 // If the slope of the colorline is zero, the gradient is vertical
103 if (clSlope == 0.0f)
104 {
105 isVertical = true;
106 }
107 else
108 {
109 // First color line intersects (x0, coord0)
110 clOffset = start.y - start.x * clSlope;
111
112 // Pixel distance between first and last color line horizontally
113 const float transformed_cvx = end.x - start.x;
114 const float transformed_cvy = end.y - start.y;
115 horizontalDistance = transformed_cvx - transformed_cvy / clSlope;
116
117 // Color increment when moving 1 pixel horizontally
118 deltaColor = 1024.0f / horizontalDistance;
119
120 if (horizontalDistance < 0)
121 {
122 horizontalDistance = -horizontalDistance;
123 }
124 return;
125 }
126 }
127 }
128
129 if (isVertical)
130 {
131 // Color change over one line
132 deltaColor = 1023.9999f / (end.y - start.y);
133
134 // Save y-coordinates for the vertical gradient.
135 // With coord0 lowest y coordinate, coord1 highest
136 if (start.y < end.y)
137 {
138 coord0 = static_cast<int16_t>(start.y);
139 coord1 = static_cast<int16_t>(end.y);
140 }
141 else
142 {
143 coord0 = static_cast<int16_t>(end.y);
144 coord1 = static_cast<int16_t>(start.y);
145 }
146 }
147 else if (isHorizontal)
148 {
149 // Color change over one line
150 deltaColor = 1023.9999f / (end.x - start.x);
151
152 // Save x-coordinates for the horizontal gradient.
153 // With x0 lowest x coordinate, x1 highest
154 if (start.x < end.x)
155 {
156 coord0 = static_cast<int16_t>(start.x);
157 coord1 = static_cast<int16_t>(end.x);
158 }
159 else
160 {
161 coord0 = static_cast<int16_t>(end.x);
162 coord1 = static_cast<int16_t>(start.x);
163 }
164 horizontalDistance = static_cast<float>(coord1 - coord0);
165 }
166 }
