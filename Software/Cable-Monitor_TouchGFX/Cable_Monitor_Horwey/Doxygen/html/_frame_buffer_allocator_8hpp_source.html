159 virtual uint16_t allocateBlock(const uint16_t x, const uint16_t y, const uint16_t width, const uint16_t height, uint8_t** block)
160 {
161 drawingBlock++;
162 if (drawingBlock == blocks)
163 {
164 drawingBlock = 0;
165 }
166 while (state[drawingBlock] != EMPTY)
167 {
168 FrameBufferAllocatorWaitOnTransfer();
169 }
170 assert(state[drawingBlock] == EMPTY);
171 state[drawingBlock] = ALLOCATED;
172 const int32_t stride = width * bytes_pr_pixel;
173 const int32_t lines = block_size / stride;
174 *block = (uint8_t*)&memory[drawingBlock][0];
175 blockRect[drawingBlock].x = x;
176 blockRect[drawingBlock].y = y;
177 blockRect[drawingBlock].width = width;
178 blockRect[drawingBlock].height = MIN(height, lines);
179 return blockRect[drawingBlock].height;
180 }
