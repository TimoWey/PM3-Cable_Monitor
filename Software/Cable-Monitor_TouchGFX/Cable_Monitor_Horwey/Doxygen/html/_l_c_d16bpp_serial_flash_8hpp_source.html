474 FORCE_INLINE_FUNCTION static uint16_t bilinearInterpolate565(uint16_t c00, uint16_t c10, uint16_t c01, uint16_t c11, uint8_t x, uint8_t y)
475 {
476 assert(x < 16 && y < 16);
477 uint32_t a00 = expandRgb565(c00);
478 uint32_t a10 = expandRgb565(c10);
479 uint32_t a01 = expandRgb565(c01);
480 uint32_t a11 = expandRgb565(c11);
481
482 uint8_t xy = (x * y) >> 3;
483 return compactRgb565((a00 * (32 - 2 * y - 2 * x + xy) + a10 * (2 * x - xy) + a01 * (2 * y - xy) + a11 * xy) >> 5);
484 }
485
486 FORCE_INLINE_FUNCTION static uint16_t bilinearInterpolate565(uint16_t c00, uint16_t c10, uint8_t x)
487 {
488 assert(x < 16);
489 uint32_t a00 = expandRgb565(c00);
490 uint32_t a10 = expandRgb565(c10);
491
492 return compactRgb565((a00 * (32 - 2 * x) + a10 * (2 * x)) >> 5);
493 }

495 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t x)
496 {
497 assert(x < 16);
498 uint16_t xy10 = 16 * x;
499 uint16_t xy00 = 256 - xy10;
500
501 return (c00 * xy00 + c10 * xy10) >> 8;
502 }
503
504 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t c01, uint8_t c11, uint8_t x, uint8_t y)
505 {
506 assert(x < 16 && y < 16);
507 uint16_t xy11 = x * y;
508 uint16_t xy10 = 16 * x - xy11;
509 uint16_t xy01 = 16 * y - xy11;
510 uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
511
512 return (c00 * xy00 + c10 * xy10 + c01 * xy01 + c11 * xy11) >> 8;
513 }
514
515 FORCE_INLINE_FUNCTION static uint32_t bilinearInterpolate888(uint32_t c00, uint32_t c10, uint8_t x)
516 {
517 assert(x < 16);
518 uint16_t xy10 = 16 * x;
519 uint16_t xy00 = 256 - xy10;
520
521 return ((((c00 & 0xFF00FF) * xy00 + (c10 & 0xFF00FF) * xy10) >> 8) & 0xFF00FF) |
522 ((((c00 & 0x00FF00) * xy00 + (c10 & 0x00FF00) * xy10) >> 8) & 0x00FF00);
523 }

525 FORCE_INLINE_FUNCTION static uint32_t bilinearInterpolate888(uint32_t c00, uint32_t c10, uint32_t c01, uint32_t c11, uint8_t x, uint8_t y)
526 {
527 assert(x < 16 && y < 16);
528 uint16_t xy11 = x * y;
529 uint16_t xy10 = 16 * x - xy11;
530 uint16_t xy01 = 16 * y - xy11;
531 uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
532
533 return ((((c00 & 0xFF00FF) * xy00 + (c10 & 0xFF00FF) * xy10 + (c01 & 0xFF00FF) * xy01 + (c11 & 0xFF00FF) * xy11) >> 8) & 0xFF00FF) |
534 ((((c00 & 0x00FF00) * xy00 + (c10 & 0x00FF00) * xy10 + (c01 & 0x00FF00) * xy01 + (c11 & 0x00FF00) * xy11) >> 8) & 0x00FF00);
535 }

547 FORCE_INLINE_FUNCTION static uint32_t div31rb(uint16_t val, uint8_t factor)
548 {
549 uint32_t val32 = (val & 0xF81F) * (factor >> 3);
550 return ((val32 + 0x0801 + ((val32 >> 5) & 0xF81F)) >> 5) & 0xF81F;

553 FORCE_INLINE_FUNCTION static uint32_t div31g(uint16_t val, uint8_t factor)
554 {
555 uint32_t val32 = (val & 0x07E0) * factor;
556 return ((val32 + 0x0020 + (val32 >> 8)) >> 8) & 0x07E0;
557 }
