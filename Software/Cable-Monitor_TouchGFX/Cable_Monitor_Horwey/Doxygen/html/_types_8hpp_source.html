192 bool intersect(const Rect& other) const
193 {
194 return !(x >= other.right() || right() <= other.x || y >= other.bottom() || bottom() <= other.y);
195 }

204 bool includes(const Rect& other) const
205 {
206 return other.isEmpty() || (other.x >= x && other.y >= y && other.right() <= right() && other.bottom() <= bottom());
207 }

217 Rect operator&(const Rect& other) const
218 {
219 Rect r = *this;
220 r &= other;
221 return r;
222 }
223
230 void operator&=(const Rect& other)
231 {
232 if (intersect(other))
233 {
234 int16_t newX = MAX(x, other.x);
235 int16_t newY = MAX(y, other.y);
236
237 width = MIN(right(), other.right()) - newX;
238 height = MIN(bottom(), other.bottom()) - newY;
239 x = newX;
240 y = newY;
241 }
242 else
243 {
244 x = 0;
245 y = 0;
246 width = 0;
247 height = 0;
248 }
249 }
250
257 void expandToFit(const Rect& other)
258 {
259 if (!other.isEmpty())
260 {
261 if (isEmpty())
262 {
263 x = other.x;
264 y = other.y;
265 width = other.width;
266 height = other.height;
267 }
268 else
269 {
270 int16_t newX = MIN(x, other.x);
271 int16_t newY = MIN(y, other.y);
272
273 int16_t endPointX = MAX(right(), other.right());
274 int16_t endPointY = MAX(bottom(), other.bottom());
275
276 x = newX;
277 y = newY;
278 width = endPointX - newX;
279 height = endPointY - newY;
280 }
281 }

294 void restrictTo(int16_t max_width, int16_t max_height)
295 {
296 // Limit area to the screen (0,0,HAL::WIDTH,HAL::HEIGT)
297 if (x < 0)
298 {
299 width += x;
300 x = 0; // Negative width is ok (isEmpty => true)
301 }
302 if (width > max_width - x) // right() > max_width
303 {
304 width = max_width - x;
305 }
306 if (y < 0)

310 }
311 if (height > max_height - y) // bottom() > max_height

374template <class T, uint16_t capacity>
375class Vector
376{
377public:
379 Vector()
380 : _size(0)
381 {
382 }
383
391 T& operator[](uint16_t idx)
392 {
393 return _elem[idx];
394 }
395
403 const T& operator[](uint16_t idx) const
404 {
405 return _elem[idx];
406 }
407
417 void add(T e)
418 {
419 assert(_size < capacity && "Vector capacity exceeded");
420 if (_size < capacity)
421 {
422 _elem[_size++] = e;
423 }
424 }
425
433 void remove(T e)
434 {
435 for (int i = 0; i < _size; i++)
436 {
437 if (_elem[i] == e)
438 {
439 for (int j = i; j < _size && j < capacity - 1; j++)
440 {
441 _elem[j] = _elem[j + 1];
442 }
443 _size--;
444 }
445 }
446 }
447
456 T removeAt(uint16_t index)
457 {
458 assert(index < _size);
459
460 _size--;
461 T tmp = _elem[index];
462 for (int i = index; i < _size; i++)
463 {
464 _elem[i] = _elem[i + 1];
465 }
466 return tmp;
467 }
468
475 void quickRemoveAt(uint16_t index)
476 {
477 assert(index < _size);
478
479 _size--;
480 // No need to copy element when removing the last element in the vector
481 if (index < _size)
482 {
483 _elem[index] = _elem[_size];
484 }
485 }
486
488 void reverse()
489 {
490 uint16_t a = 0;
491 uint16_t b = _size;
492 for (; a < --b; a++)
493 {
494 T tmp = _elem[a];
495 _elem[a] = _elem[b];
496 _elem[b] = tmp;
497 }
498 }
499
508 bool contains(T elem)
509 {
510 for (uint16_t i = 0; i < _size; i++)
511 {
512 if (elem == _elem[i])
513 {
514 return true;
515 }
516 }
517 return false;
518 }
519
529 uint16_t size() const
530 {
531 return _size;
532 }
533
539 bool isEmpty() const
540 {
541 return _size == 0;
542 }
543
549 uint16_t maxCapacity() const
550 {
551 return capacity;
552 }
553
558 void clear()
559 {
560 _size = 0;
561 }
562
563private:
564 T _elem[capacity];
565 uint16_t _size;
