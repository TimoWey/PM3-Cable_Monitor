165 void nextFadeAnimationStep()
166 {
167 if (fadeAnimationRunning)
168 {
169 fadeAnimationCounter++;
170 if (fadeAnimationCounter >= fadeAnimationDelay)
171 {
172 // Adjust the used animationCounter for the startup delay
173 uint32_t actualAnimationCounter = fadeAnimationCounter - fadeAnimationDelay;
174
175 int16_t newAlpha = fadeAnimationStartAlpha + (int16_t)fadeAnimationAlphaEquation(actualAnimationCounter, 0, fadeAnimationEndAlpha - fadeAnimationStartAlpha, fadeAnimationDuration);
176
177 if (T::getAlpha() != newAlpha)
178 {
179 if (newAlpha == 0)
180 {
181 // InvalidateContent before it becomes invisible
182 T::invalidateContent();
183 T::setAlpha((uint8_t)newAlpha);
184 }
185 else
186 {
187 // InvalidateContent after we are sure that it is visible
188 T::setAlpha((uint8_t)newAlpha);
189 T::invalidateContent();
190 }
191 }
192
193 if (fadeAnimationCounter >= (uint32_t)(fadeAnimationDelay + fadeAnimationDuration))
194 {
195 // End of animation
196 fadeAnimationRunning = false;
197 fadeAnimationDuration = 0;
198 Application::getInstance()->unregisterTimerWidget(this);
199
200 if (fadeAnimationEndedCallback && fadeAnimationEndedCallback->isValid())
201 {
202 fadeAnimationEndedCallback->execute(*this);
203 }
204 }
205 }
206 }
