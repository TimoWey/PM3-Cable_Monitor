310 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t c01, uint8_t c11, uint8_t x, uint8_t y)
311 {
312 assert(x < 16 && y < 16);
313 const uint16_t xy11 = x * y;
314 const uint16_t xy10 = 16 * x - xy11;
315 const uint16_t xy01 = 16 * y - xy11;
316 const uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
317
318 return (c00 * xy00 + c10 * xy10 + c01 * xy01 + c11 * xy11) >> 8;
