324 FORCE_INLINE_FUNCTION static uint32_t blendRGBwithXrgb8888(const uint8_t fgR, const uint8_t fgG, const uint8_t fgB, const uint32_t bgXRGB8888, const uint8_t fgAlpha, const uint8_t bgAlpha)
325 {
326 const uint8_t b = div255(fgB * fgAlpha + (bgXRGB8888 & 0xFF) * bgAlpha);
327 const uint8_t g = div255(fgG * fgAlpha + ((bgXRGB8888 >> 8) & 0xFF) * bgAlpha);
328 const uint8_t r = div255(fgR * fgAlpha + ((bgXRGB8888 >> 16) & 0xFF) * bgAlpha);
329 return (r << 16) | (g << 8) | b;
330 }

568 FORCE_INLINE_FUNCTION static uint16_t bilinearInterpolate565(uint16_t c00, uint16_t c10, uint16_t c01, uint16_t c11, uint8_t x, uint8_t y)
569 {
570 assert(x < 16 && y < 16);
571 uint32_t a00 = expandRgb565(c00);
572 uint32_t a10 = expandRgb565(c10);
573 uint32_t a01 = expandRgb565(c01);
574 uint32_t a11 = expandRgb565(c11);
575
576 uint8_t xy = (x * y) >> 3;
577 return compactRgb565((a00 * (32 - 2 * y - 2 * x + xy) + a10 * (2 * x - xy) + a01 * (2 * y - xy) + a11 * xy) >> 5);
578 }

580 FORCE_INLINE_FUNCTION static uint16_t bilinearInterpolate565(uint16_t c00, uint16_t c10, uint8_t x)
581 {
582 assert(x < 16);
583 uint32_t a00 = expandRgb565(c00);
584 uint32_t a10 = expandRgb565(c10);
585
586 return compactRgb565((a00 * (32 - 2 * x) + a10 * (2 * x)) >> 5);
587 }

589 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t x)
590 {
591 assert(x < 16);
592 uint16_t xy10 = 16 * x;
593 uint16_t xy00 = 256 - xy10;
594
595 return (c00 * xy00 + c10 * xy10) >> 8;
596 }

598 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t c01, uint8_t c11, uint8_t x, uint8_t y)
599 {
600 assert(x < 16 && y < 16);
601 uint16_t xy11 = x * y;
602 uint16_t xy10 = 16 * x - xy11;
603 uint16_t xy01 = 16 * y - xy11;
604 uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
605
606 return (c00 * xy00 + c10 * xy10 + c01 * xy01 + c11 * xy11) >> 8;
607 }

609 FORCE_INLINE_FUNCTION static uint32_t bilinearInterpolate888(uint32_t c00, uint32_t c10, uint8_t x)
610 {
611 assert(x < 16);
612 uint16_t xy10 = 16 * x;
613 uint16_t xy00 = 256 - xy10;
614
615 return ((((c00 & 0xFF00FF) * xy00 + (c10 & 0xFF00FF) * xy10) >> 8) & 0xFF00FF) |
616 ((((c00 & 0x00FF00) * xy00 + (c10 & 0x00FF00) * xy10) >> 8) & 0x00FF00);
617 }

619 FORCE_INLINE_FUNCTION static uint32_t bilinearInterpolate888(uint32_t c00, uint32_t c10, uint32_t c01, uint32_t c11, uint8_t x, uint8_t y)
620 {
621 assert(x < 16 && y < 16);
622 uint16_t xy11 = x * y;
623 uint16_t xy10 = 16 * x - xy11;
624 uint16_t xy01 = 16 * y - xy11;

626

628 ((((c00 & 0x00FF00) * xy00 + (c10 & 0x00FF00) * xy10 + (c01 & 0x00FF00) * xy01 + (c11 & 0x00FF00) * xy11) >> 8) & 0x00FF00);
