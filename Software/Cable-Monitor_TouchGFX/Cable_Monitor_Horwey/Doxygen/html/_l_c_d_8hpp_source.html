525 FORCE_INLINE_FUNCTION void drawTextureMapNextSubdivision(float& ULeft, float& VLeft, float& ZRight, float& URight, float& VRight, float& oneOverZRight, const float dOneOverZdXAff, float& UOverZRight, const float dUOverZdXAff, float& VOverZRight, const float dVOverZdXAff, const int affineLength, fixed16_16& U, fixed16_16& V, fixed16_16& deltaU, fixed16_16& deltaV)
526 {
527 ULeft = URight;
528 VLeft = VRight;
529
530 oneOverZRight += dOneOverZdXAff;
531 UOverZRight += dUOverZdXAff;
532 VOverZRight += dVOverZdXAff;
533
534 ZRight = 1 / oneOverZRight;
535 URight = ZRight * UOverZRight;
536 VRight = ZRight * VOverZRight;
537
538 U = floatToFixed16_16(ULeft);
539 V = floatToFixed16_16(VLeft);
540 deltaU = floatToFixed16_16(URight - ULeft) / affineLength;
541 deltaV = floatToFixed16_16(VRight - VLeft) / affineLength;
