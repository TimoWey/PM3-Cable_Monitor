367 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t x)
368 {
369 assert(x < 16);
370 return (c00 * (16 - x) + c10 * x) >> 4;
371 }
372
373 FORCE_INLINE_FUNCTION static uint8_t bilinearInterpolate8(uint8_t c00, uint8_t c10, uint8_t c01, uint8_t c11, uint8_t x, uint8_t y)
374 {
375 assert(x < 16 && y < 16);
376 const uint16_t xy11 = x * y;
377 const uint16_t xy10 = 16 * x - xy11;
378 const uint16_t xy01 = 16 * y - xy11;
379 const uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
380
381 return (c00 * xy00 + c10 * xy10 + c01 * xy01 + c11 * xy11) >> 8;
382 }
383
384 FORCE_INLINE_FUNCTION static uint32_t bilinearInterpolate888(uint32_t c00, uint32_t c10, uint8_t x)
385 {
386 assert(x < 16);
387 const uint16_t xy00 = 16 - x;
388
389 return ((((c00 & 0xFF00FF) * xy00 + (c10 & 0xFF00FF) * x) >> 4) & 0xFF00FF) |
390 ((((c00 & 0x00FF00) * xy00 + (c10 & 0x00FF00) * x) >> 4) & 0x00FF00);
391 }
392
393 FORCE_INLINE_FUNCTION static uint32_t bilinearInterpolate888(uint32_t c00, uint32_t c10, uint32_t c01, uint32_t c11, uint8_t x, uint8_t y)
394 {
395 assert(x < 16 && y < 16);
396 const uint16_t xy11 = x * y;
397 const uint16_t xy10 = 16 * x - xy11;
398 const uint16_t xy01 = 16 * y - xy11;
399 const uint16_t xy00 = 256 - (xy11 + xy10 + xy01);
400
401 return ((((c00 & 0xFF00FF) * xy00 + (c10 & 0xFF00FF) * xy10 + (c01 & 0xFF00FF) * xy01 + (c11 & 0xFF00FF) * xy11) >> 8) & 0xFF00FF) |
402 ((((c00 & 0x00FF00) * xy00 + (c10 & 0x00FF00) * xy10 + (c01 & 0x00FF00) * xy01 + (c11 & 0x00FF00) * xy11) >> 8) & 0x00FF00);
403 }

405 FORCE_INLINE_FUNCTION static uint32_t div255_888(uint32_t val, uint8_t factor)
406 {
407 return div255rb((val & 0xFF00FF) * factor) | div255g((val & 0x00FF00) * factor);
408 }

410 FORCE_INLINE_FUNCTION static uint32_t div255_888_FFcheck(uint32_t val, uint8_t factor)
411 {
412 return factor < 0xFF ? div255_888(val, factor) : val;
413 }
