93 static void getRGBFromHSV(uint8_t hue, uint8_t saturation, uint8_t value, uint8_t& red, uint8_t& green, uint8_t& blue)
94 {
95 if (saturation == 0)
96 {
97 red = green = blue = value;
98 return;
99 }
100
101 const uint8_t region = (hue * 6) >> 8;
102 const int next_region_start = ((((region + 1) << 8) + 5) / 6); // Can go up to 256, uint8_t not enough
103 const uint8_t region_size = next_region_start - (((region << 8) + 5) / 6);
104 const uint8_t remainder = 255 - (next_region_start - hue) * 255 / region_size;
105
106 const uint8_t p = LCD::div255(value * (255 - saturation));
107 const uint8_t q = LCD::div255(value * (255 - LCD::div255(saturation * remainder)));
108 const uint8_t t = LCD::div255(value * (255 - LCD::div255(saturation * (255 - remainder))));
109
110 switch (region)
111 {
112 case 0:
113 red = value;
114 green = t;
115 blue = p;
116 break;
117 case 1:
118 red = q;
119 green = value;
120 blue = p;
121 break;
122 case 2:
123 red = p;
124 green = value;
125 blue = t;
126 break;
127 case 3:
128 red = p;
129 green = q;
130 blue = value;
131 break;
132 case 4:
133 red = t;
134 green = p;
135 blue = value;
136 break;
137 default:
138 red = value;
139 green = p;
140 blue = q;
141 break;
142 }
143 }

157 static void getHSVFromRGB(uint8_t red, uint8_t green, uint8_t blue, uint8_t& hue, uint8_t& saturation, uint8_t& value)
158 {
159 const uint8_t rgbMin = MIN(MIN(red, green), blue);
160 const uint8_t rgbMax = MAX(MAX(red, green), blue);
161 const uint8_t rgbRange = rgbMax - rgbMin;
162
163 value = rgbMax;
164 if (value == 0)
165 {
166 hue = 0;
167 saturation = 0;
168 return;
169 }
170
171 saturation = 255 * rgbRange / value;
172 if (saturation == 0)
173 {
174 hue = 0;
175 }
176 else if (rgbMax == red)
177 {
178 if (green < blue)
179 {
180 hue = 0 + (42 * (green - blue) + rgbRange / 2) / rgbRange; // [0-42; 0] = [214; 0]
181 }
182 else
183 {
184 hue = 0 + (43 * (green - blue) + rgbRange / 2) / rgbRange; // [0; 0+43] = [0; 43]
185 }
186 }
187 else if (rgbMax == green)
188 {
189 if (blue < red)
190 {
191 hue = 86 + (43 * (blue - red) + rgbRange / 2) / rgbRange; // [86-43; 86] = [43; 86]
192 }
193 else
194 {
195 hue = 86 + (42 * (blue - red) + rgbRange / 2) / rgbRange; // [86; 86+42] = [86; 128]
196 }
197 }
198 else
199 {
200 hue = 171 + (43 * (red - green) + rgbRange / 2) / rgbRange; // [171-43; 171+43] = [128; 214]
201 }
202 }

261 FORCE_INLINE_FUNCTION static void getHSVFromHSL(uint8_t hsl_hue, uint8_t hsl_saturation, uint8_t hsl_luminance, uint8_t& hsv_hue, uint8_t& hsv_saturation, uint8_t& hsv_value)
262 {
263 hsv_hue = hsl_hue;
264 hsv_value = hsl_luminance + LCD::div255(hsl_saturation * MIN(hsl_luminance, 255 - hsl_luminance));
265 hsv_saturation = hsv_value == 0 ? 0 : 510 - ((hsl_luminance * 510) / hsv_value);
266 }

278 FORCE_INLINE_FUNCTION static void getHSLFromHSV(uint8_t hsv_hue, uint8_t hsv_saturation, uint8_t hsv_value, uint8_t& hsl_hue, uint8_t& hsl_saturation, uint8_t& hsl_luminance)
279 {
280 hsl_hue = hsv_hue;
281 uint16_t luminance2 = (hsv_value * (510 - hsv_saturation)) / 255;
282 hsl_luminance = luminance2 >> 1;
283 hsl_saturation = (hsl_luminance == 0 || hsl_luminance == 255) ? 0 : (uint8_t)(((hsv_value * 2 - luminance2) * 255) / MIN(luminance2, 510 - luminance2));
284 }
