47 SlideTransition(const uint8_t transitionSteps = 20)
48 : Transition(),
49 snapshot(),
50 animationSteps(transitionSteps),
51 animationCounter(0),
52 relativeValue(0)
53 {
54 if (!HAL::USE_ANIMATION_STORAGE)
55 {
56 // No animation storage, don't slide
57 done = true;
58 return;
59 }
60
61 snapshot.setPosition(0, 0, HAL::DISPLAY_WIDTH, HAL::DISPLAY_HEIGHT);
62 snapshot.makeSnapshot();
63
64 switch (templateDirection)
65 {
66 case EAST:
67 targetValue = -HAL::DISPLAY_WIDTH;
68 break;
69 case WEST:
70 targetValue = HAL::DISPLAY_WIDTH;
71 break;
72 case NORTH:
73 targetValue = HAL::DISPLAY_HEIGHT;
74 break;
75 case SOUTH:
76 targetValue = -HAL::DISPLAY_HEIGHT;
77 break;
78 }
79 }
80
87 virtual void handleTickEvent()
88 {
89 Transition::handleTickEvent();
90
91 // Calculate new position or stop animation
92 animationCounter++;
93 if (animationCounter > animationSteps)
94 {
95 // Final step: stop the animation
96 done = true;
97 animationCounter = 0;
98 return;
99 }
100
101 // Calculate value in [0;targetValue]
102 int16_t calculatedValue = EasingEquations::cubicEaseOut(animationCounter, 0, targetValue, animationSteps);
103
104 // Note: Result of "calculatedValue & 1" is compiler dependent for negative values of calculatedValue
105 if ((calculatedValue % 2) != 0)
106 {
107 // Optimization: calculatedValue is odd, add 1/-1 to move drawables modulo 32 bits in framebuffer
108 calculatedValue += (calculatedValue > 0 ? 1 : -1);
109 }
110
111 // Move snapshot
112 switch (templateDirection)
113 {
114 case EAST:
115 case WEST:
116 relativeValue = calculatedValue - snapshot.getX();
117 break;
118 case NORTH:
119 case SOUTH:
120 relativeValue = calculatedValue - snapshot.getY();
121 break;
122 }
123
124 // Move snapshot and its children with delta value for X or Y
125 Drawable* d = screenContainer->getFirstChild();
126 while (d)
127 {
128 switch (templateDirection)
129 {
130 case EAST:
131 case WEST:
132 d->setX(d->getX() + relativeValue);
133 break;
134 case NORTH:
135 case SOUTH:
136 d->setY(d->getY() + relativeValue);
137 break;
138 }
139 d = d->getNextSibling();
140 }
141
142 // Entire screen has changed, redraw
143 screenContainer->invalidate();
144 }
145
146 virtual void tearDown()
147 {
148 if (HAL::USE_ANIMATION_STORAGE && screenContainer)
149 {
150 screenContainer->remove(snapshot);
151 }
152 }
153
154 virtual void init()
155 {
156 Transition::init();
157
