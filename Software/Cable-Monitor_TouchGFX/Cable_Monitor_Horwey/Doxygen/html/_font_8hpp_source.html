88 FORCE_INLINE_FUNCTION int16_t top() const
89 {

100 FORCE_INLINE_FUNCTION void setTop(int16_t newTop)
101 {
102 _top = newTop & 0xFF;
103 flags &= ~(GLYPH_DATA_TOP_BIT8 | GLYPH_DATA_TOP_BIT9);
104 flags |= (newTop & 0x300) >> 3;
105 }

113 {
114 return ((flags & GLYPH_DATA_ADVANCE_BIT8) << 1) | _advance;
