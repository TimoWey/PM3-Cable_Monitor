139/**
140 * @brief Converts a float variable to an array of bytes.
141 *
142 * This function takes a float variable and converts it into an array of bytes.
143 * It uses a union to access the individual bytes of the float variable.
144 * The resulting byte array is stored in the provided ftoa_bytes_temp array.
145 *
146 * @param ftoa_bytes_temp Pointer to the array where the byte array will be stored.
147 * @param float_variable The float variable to be converted.
148 */
149void float2Bytes(uint8_t * ftoa_bytes_temp, float float_variable)
150{
151 union {
152 float a;
153 uint8_t bytes[4];
154 } thing;
155
156 thing.a = float_variable;
157
158 for (uint8_t i = 0; i < 4; i++) {
159 ftoa_bytes_temp[i] = thing.bytes[i];
160 }
161}

187/**
188 * @brief Writes data to the specified flash sector.
189 * @param StartSectorAddress: The starting address of the flash sector.
190 * @param Data: Pointer to the data to be written.
191 * @param numberofwords: Number of words to be written.
192 * @return HAL_FLASH_GetError() if an error occurred during erasing or writing the flash sector, otherwise 0.
193 */
194uint32_t Flash_Write_Data (uint32_t StartSectorAddress, uint32_t *Data, uint16_t numberofwords)
195{
196
197 static FLASH_EraseInitTypeDef EraseInitStruct;
198 uint32_t SECTORError;
199 int sofar=0;
200
201
202 /* Unlock the Flash to enable the flash control register access *************/
203 HAL_FLASH_Unlock();
204
205 /* Erase the user Flash area */
206
207 /* Get the number of sector to erase from 1st sector */
208
209 uint32_t StartSector = GetSector(StartSectorAddress);
210 uint32_t EndSectorAddress = StartSectorAddress + numberofwords*4;
211 uint32_t EndSector = GetSector(EndSectorAddress);
212
213 /* Fill EraseInit structure*/
214 EraseInitStruct.TypeErase = FLASH_TYPEERASE_SECTORS;
215 EraseInitStruct.VoltageRange = FLASH_VOLTAGE_RANGE_3;
216 EraseInitStruct.Sector = StartSector;
217 EraseInitStruct.NbSectors = (EndSector - StartSector) + 1;
218
219 /* Note: If an erase operation in Flash memory also concerns data in the data or instruction cache,
220 you have to make sure that these data are rewritten before they are accessed during code
221 execution. If this cannot be done safely, it is recommended to flush the caches by setting the
222 DCRST and ICRST bits in the FLASH_CR register. */
223 if (HAL_FLASHEx_Erase(&EraseInitStruct, &SECTORError) != HAL_OK)
224 {
225 return HAL_FLASH_GetError ();
226 }
227
228 /* Program the user Flash area word by word
229 (area defined by FLASH_USER_START_ADDR and FLASH_USER_END_ADDR) ***********/
230
231 while (sofar<numberofwords)
232 {
233 if (HAL_FLASH_Program(FLASH_TYPEPROGRAM_WORD, StartSectorAddress, Data[sofar]) == HAL_OK)
234 {
235 StartSectorAddress += 4; // use StartPageAddress += 2 for half word and 8 for double word
236 sofar++;
237 }
238 else
239 {
240 /* Error occurred while writing data in Flash memory*/
241 return HAL_FLASH_GetError ();
242 }
243 }
244
245 /* Lock the Flash to disable the flash control register access (recommended
246 to protect the FLASH memory against possible unwanted operation) *********/
247 HAL_FLASH_Lock();
248
249 return 0;
250}

252/**
253 * @brief Reads data from flash memory.
254 *
255 * This function reads data from the specified flash memory sector starting at the given address.
256 * The data is stored in the provided buffer.
257 *
258 * @param StartSectorAddress The starting address of the flash memory sector.
259 * @param RxBuf Pointer to the buffer where the read data will be stored.
260 * @param numberofwords The number of words (32-bit) to read from the flash memory sector.
261 */
262void Flash_Read_Data (uint32_t StartSectorAddress, uint32_t *RxBuf, uint16_t numberofwords)
263{
264 while (1)
265 {
266 *RxBuf = *(__IO uint32_t *)StartSectorAddress;
267 StartSectorAddress += 4;
268 RxBuf++;
269 if (!(numberofwords--)) break;
270 }
271}

273/**
274 * @brief Converts an array of 32-bit data into a string representation.
275 *
276 * This function takes an array of 32-bit data and converts it into a string representation.
277 * The resulting string is stored in the provided buffer.
278 *
279 * @param Data Pointer to the array of 32-bit data.
280 * @param Buf Pointer to the buffer where the resulting string will be stored.
281 */
282void Convert_To_Str (uint32_t *Data, char *Buf)
283{
284 int numberofbytes = ((strlen((char *)Data)/4) + ((strlen((char *)Data) % 4) != 0)) *4;
285
286 for (int i=0; i<numberofbytes; i++)
287 {
288 Buf[i] = Data[i/4]>>(8*(i%4));
289 }
290}
