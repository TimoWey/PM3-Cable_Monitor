171void MEAS_ADC3_scan_init(void)
172{
173 __HAL_RCC_ADC3_CLK_ENABLE(); // Enable Clock for ADC3
174 ADC3->SQR1 |= (3UL << ADC_SQR1_L_Pos); // Convert 4 inputs (4-1)
175 ADC3->SQR3 |= (4UL << ADC_SQR3_SQ1_Pos); // Input 4 = first conversion
176 ADC3->SQR3 |= (13UL << ADC_SQR3_SQ2_Pos); // Input 13 = second conversion
177 ADC3->SQR3 |= (6UL << ADC_SQR3_SQ3_Pos); // Input 6 = third conversion
178 ADC3->SQR3 |= (11UL << ADC_SQR3_SQ4_Pos); // Input 11 = fourth conversion
179 ADC3->CR1 |= ADC_CR1_SCAN; // Enable scan mode
180 ADC3->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
181 ADC3->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
182 ADC3->CR2 |= ADC_CR2_DMA; // Enable DMA mode
183 __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
184 // set DMA to highest priority
185 NVIC_SetPriority(DMA2_Stream1_IRQn, 0);
186 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 1
187 while (DMA2_Stream1->CR & DMA_SxCR_EN)
188 {
189
190 } // Wait for DMA to finish
191 DMA2->LIFCR |= DMA_LIFCR_CTCIF1; // Clear transfer complete interrupt fl.
192 DMA2_Stream1->CR |= (2UL << DMA_SxCR_CHSEL_Pos); // Select channel 2
193 DMA2_Stream1->CR |= DMA_SxCR_PL_0; // Priority high
194 DMA2_Stream1->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
195 DMA2_Stream1->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
196 DMA2_Stream1->CR |= DMA_SxCR_MINC; // Increment memory address pointer
197 DMA2_Stream1->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
198 // Number of data items to transfer
199 DMA2_Stream1->NDTR = INPUT_COUNT * ADC_NUMS;
200 DMA2_Stream1->PAR = (uint32_t)&ADC3->DR; // Peripheral register address
201 DMA2_Stream1->M0AR = (uint32_t)ADC_samples; // Buffer memory loc. address
202}

208void MEAS_ADC3_scan_start(void)
209{

211 NVIC_ClearPendingIRQ(DMA2_Stream1_IRQn); // Clear pending DMA interrupt
212 NVIC_EnableIRQ(DMA2_Stream1_IRQn); // Enable DMA interrupt in the NVIC

214 TIM2->CR1 |= TIM_CR1_CEN; // Enable timer
