1/** *****************************************************************************
2 * @file
3 * @brief Measuring input signal using the following configuration:
4 *
5 * Measurement Procedure
6 * -----------
7 * @details The measurement consists of an amplified signal from the cable monitor board.
8 * This signal is then sampled by the ADC and the samples are stored in a buffer.
9 * The buffer is then processed by the calculation module using several digital signal processing methods.
10 * @note This processing allows the calculation of the distance to the cable, the angle of the cable, and the current flowing through the cable.
11 * @n Refer to @ref calculation.c
12 *
13 * The simplified procedure is shown in the following figure:
14 * @image html Doxygen/img/ADC_Process.png height=600px
15 *
16 *
17 * Configuration of ADC, Timer, and DMA
18 * -----------
19 * - ADC is triggered by a timer and with an interrupt after the end of conversion
20 * - ADC combined with DMA (Direct Memory Access) to fill a buffer
21 * - Scan mode = sequential sampling of two inputs by one ADC
22 * - Analog mode configuration for GPIOs
23 *
24 * @details ADC3 is configured to sample four channels sequentially.
25 * @n The ADC is triggered by Timer 2, and the samples are transferred to memory by the DMA.
26 * @n The DMA triggers the transfer complete interrupt when all data is ready.
27 * @n The inputs used are ADC3_IN4 = GPIO PF6 (Pad Right), ADC3_IN13 = GPIO PC3 (Pad Left),
28 * ADC3_IN6 = GPIO PF8 (HS-Right), and ADC3_IN11 = GPIO PC1 (HS-Left)
29 * @n Refer to @ref MEAS_start_measure
30 * @image html Doxygen/img/DMA_Buffer.png height=800px
31 *
32 * @note The following peripherals are used:
33 * - ADC3
34 * - DMA2
35 * - TIM2
36 * - GPIO
37 * - NVIC
38 * - RCC
39 *
40 * @anchor HowTo
41 * How to Configure the Peripherals: ADC, Timer, and DMA
42 * ====================================================
43 *
44 * All the peripherals are accessed by writing to or reading from registers.
45 * From the programmer’s point of view, this is done exactly as
46 * writing or reading the value of a variable.
47 * @n Writing to a register configures the HW of the associated peripheral
48 * to do what is required.
49 * @n Reading from a register gets status and data from the HW peripheral.
50 *
51 * The information on which bits have to be set to get a specific behavior
52 * is documented in the
53 * @htmlonly
54 * <a href="../files/Reference manual STM32F429.pdf">reference manual</a>
55 * @endhtmlonly of the microcontroller.
56 *

96/** ***************************************************************************
97 * @brief Configure GPIOs in analog mode.
98 *
99 * @note The input number for the ADCs is not equal to the GPIO pin number!
100 * - ADC3_IN4 = GPIO PF6 (Pad right)
101 * - ADC123_IN13 = GPIO PC3 (Pad left)
102 * - ADC3_IN6 = GPIO PF8 (Hall sensor right)
103 * - ADC123_IN11 = GPIO PC1 (Hall sensor left)
104 *****************************************************************************/
105void MEAS_GPIO_analog_init(void)
106{
107 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
108 GPIOF->MODER |= (3UL << GPIO_MODER_MODER6_Pos); // Analog PF6 = ADC3_IN4
109 GPIOF->MODER |= (3UL << GPIO_MODER_MODER8_Pos); // Analog PF8 = ADC3_IN6
110 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
111 GPIOC->MODER |= (3UL << GPIO_MODER_MODER3_Pos); // Analog PC3 = ADC3_IN13
112 GPIOC->MODER |= (3UL << GPIO_MODER_MODER1_Pos); // Analog PC1 = ADC3_IN11
113}

115/**
116 * @brief Resets the ADCs, disables the timer.
117 *
118 * This function resets the ADCs by asserting the ADC reset bit in the RCC_APB2RSTR register,
119 * then releases the reset by clearing the ADC reset bit. It also disables the timer by clearing
120 * the TIM_CR1_CEN bit in the TIM2 control register.
121 */
122void MEAS_ADC_reset(void)
123{
124 RCC->APB2RSTR |= RCC_APB2RSTR_ADCRST; // Reset ADCs
125 RCC->APB2RSTR &= ~RCC_APB2RSTR_ADCRST; // Release reset of ADCs
126 TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
127}

143/** ***************************************************************************
144 * @brief Configure the timer to trigger the ADC(s)
145 *
146 * @note For debugging purposes the timer interrupt might be useful.
147 *****************************************************************************/
148void MEAS_timer_init(void)
149{
150 __HAL_RCC_TIM2_CLK_ENABLE(); // Enable Clock for TIM2
151 TIM2->PSC = TIM_PRESCALE; // Prescaler for clock freq. = 1MHz
152 TIM2->ARR = TIM_TOP; // Auto reload = counter top value
153 TIM2->CR2 |= TIM_CR2_MMS_1; // TRGO on update
154
155 // set timer to highest priority
156 NVIC_SetPriority(TIM2_IRQn, 0);
157}

159/** ***************************************************************************
160 * @brief Initialize ADC, timer and DMA for sequential acquisition = scan mode
161 *
162 * Uses ADC2 and DMA2_Stream1 channel2
163 * @n The ADC2 trigger is set to TIM2 TRGO event
164 * @n At each trigger both inputs are converted sequentially
165 * and transfered to memory by the DMA.
166 * @n As each conversion triggers the DMA, the number of transfers is doubled.
167 * @n The DMA triggers the transfer complete interrupt when all data is ready.
168 * @n The inputs used are ADC3_IN4 = GPIO PF6 (Pad Right), ADC3_IN13 = GPIO PC3
169 *(Pad Left), ADC_IN6 = GPIO PF8 (HS-Right), ADC3_IN11 = GPIO PC1 (HS-Left)
170 *****************************************************************************/
171void MEAS_ADC3_scan_init(void)
172{
173 __HAL_RCC_ADC3_CLK_ENABLE(); // Enable Clock for ADC3
174 ADC3->SQR1 |= (3UL << ADC_SQR1_L_Pos); // Convert 4 inputs (4-1)
175 ADC3->SQR3 |= (4UL << ADC_SQR3_SQ1_Pos); // Input 4 = first conversion
176 ADC3->SQR3 |= (13UL << ADC_SQR3_SQ2_Pos); // Input 13 = second conversion
177 ADC3->SQR3 |= (6UL << ADC_SQR3_SQ3_Pos); // Input 6 = third conversion
178 ADC3->SQR3 |= (11UL << ADC_SQR3_SQ4_Pos); // Input 11 = fourth conversion
179 ADC3->CR1 |= ADC_CR1_SCAN; // Enable scan mode
180 ADC3->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
181 ADC3->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
182 ADC3->CR2 |= ADC_CR2_DMA; // Enable DMA mode
183 __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
184 // set DMA to highest priority
185 NVIC_SetPriority(DMA2_Stream1_IRQn, 0);
186 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 1
187 while (DMA2_Stream1->CR & DMA_SxCR_EN)
188 {
189
190 } // Wait for DMA to finish
191 DMA2->LIFCR |= DMA_LIFCR_CTCIF1; // Clear transfer complete interrupt fl.
192 DMA2_Stream1->CR |= (2UL << DMA_SxCR_CHSEL_Pos); // Select channel 2
193 DMA2_Stream1->CR |= DMA_SxCR_PL_0; // Priority high
194 DMA2_Stream1->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
195 DMA2_Stream1->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
196 DMA2_Stream1->CR |= DMA_SxCR_MINC; // Increment memory address pointer
197 DMA2_Stream1->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
198 // Number of data items to transfer
199 DMA2_Stream1->NDTR = INPUT_COUNT * ADC_NUMS;
200 DMA2_Stream1->PAR = (uint32_t)&ADC3->DR; // Peripheral register address
201 DMA2_Stream1->M0AR = (uint32_t)ADC_samples; // Buffer memory loc. address
202}

204/** ***************************************************************************
205 * @brief Start DMA, ADC and timer
206 *
207 *****************************************************************************/
208void MEAS_ADC3_scan_start(void)
209{
210 DMA2_Stream1->CR |= DMA_SxCR_EN; // Enable DMA
211 NVIC_ClearPendingIRQ(DMA2_Stream1_IRQn); // Clear pending DMA interrupt
212 NVIC_EnableIRQ(DMA2_Stream1_IRQn); // Enable DMA interrupt in the NVIC
213 ADC3->CR2 |= ADC_CR2_ADON; // Enable ADC3
214 TIM2->CR1 |= TIM_CR1_CEN; // Enable timer
215}

217/** ***************************************************************************
218 * @brief Interrupt handler for DMA2 Stream1
219 *
220 * The samples from the ADC3 have been transfered to memory by the DMA2 Stream1
221 * and are ready for processing.
222 *****************************************************************************/
223void DMA2_Stream1_IRQHandler(void)
224{
225 if (DMA2->LISR & DMA_LISR_TCIF1) // Stream1 transfer compl. interrupt f.
226 {
227 // Disable DMA interrupt in the NVIC
228 NVIC_DisableIRQ(DMA2_Stream1_IRQn);
229 NVIC_ClearPendingIRQ(DMA2_Stream1_IRQn); // clr pending DMA interrupt
230 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA
231 while (DMA2_Stream1->CR & DMA_SxCR_EN) // Wait for DMA to finish
232 {
233
234 }
235
236 DMA2->LIFCR |= DMA_LIFCR_CTCIF1;// clr transfer complete interrupt fl.
237 TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
238 ADC3->CR2 &= ~ADC_CR2_ADON; // Disable ADC3
239 ADC3->CR2 &= ~ADC_CR2_DMA; // Disable DMA mode
240
241 MEAS_ADC_reset();
242 }

249 * @brief Start the measurement by initialising the ADC, timer and DMA, it then

257uint32_t* MEAS_start_measure(void)
258{
259 MEAS_Buffer_reset(INPUT_COUNT, ADC_samples);
260 MEAS_ADC3_scan_init();
261 MEAS_ADC3_scan_start();
262
263 while (DMA2_Stream1->CR & DMA_SxCR_EN) // Wait for DMA to finish
264 {
265
266 }
267
268 return ADC_samples;
