146FFT calculate_freq_and_signalstrength(float32_t input_samples[])
147{
148 // remove DC component
149 for (uint32_t i = 0; i < SAMPLE_LEN; i++) {
150 input_samples[i] -= 2047.5;
151 }
152 // Create an array for the FFT output
153 float32_t fft_complex_output[2 * SAMPLE_LEN];
154 // Create an instance of the FFT structure
155 arm_rfft_fast_instance_f32 fft_struct;
156 // Initialize the FFT structure
157 arm_rfft_fast_init_f32(&fft_struct, SAMPLE_LEN);
158 // Perform the FFT on the input samples
159 arm_rfft_fast_f32(&fft_struct, input_samples, fft_complex_output, 0);
160 // Calculate magnitude of complex numbers
161 arm_cmplx_mag_f32(fft_complex_output, fft_complex_output, SAMPLE_LEN / 2);
162 // Find the index of the maximum magnitude in the first half of the array
163 uint32_t maxIndex;
164 float32_t maxValue;
165 arm_max_f32(fft_complex_output, SAMPLE_LEN / 2, &maxValue, &maxIndex);
166 // Create an instance of the FFT structure
167 FFT fft;
168 // Calculate the main frequency
169 float main_frequency =
170 (float)maxIndex * ((float)MEAS_get_samp_freq()) / SAMPLE_LEN;
171 uint8_t given_frequency = 0;
172 // Check if the main frequency is in the range of 45-55 Hz
173 if (main_frequency <= 55 && main_frequency >= 45)
174 {
175 // If yes, round the given frequency to 50 Hz
176 given_frequency = 50;
177 }
178 else if (main_frequency <= 65 && main_frequency >= 55)
179 {
180 // If yes, round the given frequency to 60 Hz
181 given_frequency = 60;
182 }
183 else
184 given_frequency = 0;
185 // calculate the index of the given frequency
186 uint32_t index = (uint32_t)given_frequency * SAMPLE_LEN / MEAS_get_samp_freq();
187 // set the signal strength to 0
188 float signal_strength = 0;
189 // check if the given frequency is 50 or 60 Hz
190 if (given_frequency != 0)
191 {
192 // calculate the signal strength
193 signal_strength = 2 * fft_complex_output[index] / SAMPLE_LEN;
194 fft.error = CALC_ERROR_NONE;
195 }
196 else
197 fft.error = CALC_ERROR_FREQUENCY; // if the given frequency is not 50 or 60 Hz
198 // return the calculated values
199 fft.main_freq = main_frequency;
200 fft.signal_strength = signal_strength;
201 return fft;
202}

285SINGLE_MEAS single_measurement(uint8_t Phase)
286{
287 // Start the measurement
288 uint32_t* samples = MEAS_start_measure();
289 // Create an instance of the SINGLE_MEAS structure
290 SINGLE_MEAS single_meas;
291 // Create an instance of the FFT structure
292 FFT fft;
293
294 // Set the error to none
295 single_meas.error = CALC_ERROR_NONE;
296
297 static float32_t samples_PR[SAMPLE_LEN];
298 static float32_t samples_PL[SAMPLE_LEN];
299 static float32_t samples_HSR[SAMPLE_LEN];
300 static float32_t samples_HSL[SAMPLE_LEN];
301
302 float32_t current_l = 0;
303 float32_t current_r = 0;
304
305 // Convert the samples to the specific channel
306 for (uint32_t i = 0; i < 64; i++)
307 {
308 samples_PR[i] = (float32_t)samples[(i * CHANNEL_NUM)];
309 samples_PL[i] = (float32_t)samples[(i * CHANNEL_NUM) + 1];
310 samples_HSR[i] = (float32_t)samples[(i * CHANNEL_NUM) + 2];
311 samples_HSL[i] = (float32_t)samples[(i * CHANNEL_NUM) + 3];
312 }
313 // Calculate the Signal Strength and the main frequency for each channel using FFT
314 fft = calculate_freq_and_signalstrength(samples_PR);
315 float32_t signal_strength_PR = fft.signal_strength;
316 float32_t main_frequency_PR = fft.main_freq;
317 fft = calculate_freq_and_signalstrength(samples_PL);
318 float32_t signal_strength_PL = fft.signal_strength;
319 float32_t main_frequency_PL = fft.main_freq;
320 fft = calculate_freq_and_signalstrength(samples_HSR);
321 float32_t signal_strength_HSR = fft.signal_strength;
322 fft = calculate_freq_and_signalstrength(samples_HSL);
323 float32_t signal_strength_HSL = fft.signal_strength;
324 // Calculate the main frequency
325 single_meas.frequency = (main_frequency_PR + main_frequency_PL) / 2;
326 // Calculate distance and angle
327 DISTANCE_ANGLE dist_angle = calculate_distance_and_angle(signal_strength_PR, signal_strength_PL);
328 single_meas.distance = dist_angle.distance;
329 single_meas.angle = dist_angle.angle;
330
331 // Calculate the current using the magnetic coefficients for 5A
332 if(Phase == SINGLE_PHASE)
333 {
334 current_l = a_magn2_l_1P * single_meas.distance * signal_strength_HSL;
335 current_r = a_magn2_r_1P * single_meas.distance * signal_strength_HSR;
336 single_meas.current = (current_l + current_r) / 2;
337 }
338 else if(Phase == THREE_PHASE)
339 {
340 if (signal_strength_HSL > signal_strength_HSR)
341 single_meas.current = a_magn2_l_3P * single_meas.distance * signal_strength_HSL;
342 else
343 single_meas.current = a_magn2_r_3P * single_meas.distance * signal_strength_HSR;
344 }
345
346 // Check if the current is smaller than 4 A -> use the magnetic coefficients for 1.2 A
347 if (single_meas.current < 4)
348 {
349 if(Phase == SINGLE_PHASE)
350 {
351 current_l = a_magn1_l_1P * single_meas.distance * signal_strength_HSL;
352 current_r = a_magn1_r_1P * single_meas.distance * signal_strength_HSR;
353 single_meas.current = (current_l + current_r) / 2;
354 }
355 else if(Phase == THREE_PHASE)
356 {
357 if (signal_strength_HSL > signal_strength_HSR)
358 single_meas.current = a_magn1_l_3P * single_meas.distance * signal_strength_HSL;
359 else
360 single_meas.current = a_magn1_r_3P * single_meas.distance * signal_strength_HSR;
361 }
362 }
363
364 // ERROR TESTING:
365
366 // Check if distance is close enough for current calculation and if the current is out of the range of 0-10 A
367 if (single_meas.distance > 30 || (single_meas.current <= 0 && single_meas.current >= 10))
368 single_meas.error = CALC_ERROR_TOO_FAR_AWAY;
369
370 // Check if the current is larger than 10 A -> CALL ERROR OVERCURRENT
371 if (single_meas.current > 10)
372 single_meas.error = CALC_ERROR_OVERCURRENT;
373
374 // Check if the distance is larger than 300 mm or smaller than 0 -> CALL ERROR DISCONNECT
375 if (single_meas.distance > 250)
376 {
377 single_meas.error = CALC_ERROR_DISCONNECT;
378 single_meas.angle = 0;
379 }
380
381 // Check if the frequency is 0 -> CALL ERROR FREQUENCY
382 if (single_meas.frequency == 0)
383 single_meas.error = CALC_ERROR_FREQUENCY;
384
385 // OPTIMIZE:
386 // Limit the angle to +/- 90 degrees
387 if (single_meas.angle > 90)
388 single_meas.angle = 90;
389 else if (single_meas.angle < -90)
390 single_meas.angle = -90;
391
392 // BUZZER LED
393 if(single_meas.distance > 0 && single_meas.distance < 200 && (State_BUZZER == false))
394 TIM14->CCR1 = 50 + (single_meas.distance/4);
395 else
396 TIM14->CCR1 = 0;
397
398 if(single_meas.distance > 0 && single_meas.distance < 200 && (State_LED == false))
399 TIM5->CCR1 = abs(100 - (int)(single_meas.distance/2));
400 else
401 TIM5->CCR1 = 0;
402 // Return the calculated values
403 return single_meas;
404}

417ACCU_MEAS accurate_measurement(uint8_t Phase)
418{
419 // Create an instance of the ACCU_MEAS structure
420 ACCU_MEAS accu_meas;
421
422 // set the error to none
423 accu_meas.error_accu = CALC_ERROR_NONE;
424
425 static float32_t distance[ACCURATE_MEASUREMENT_LOOPS];
426 static float32_t angle[ACCURATE_MEASUREMENT_LOOPS];
427 static float32_t frequency[ACCURATE_MEASUREMENT_LOOPS];
428 static float32_t current[ACCURATE_MEASUREMENT_LOOPS];
429
430 for(uint8_t i = 0; i < ACCURATE_MEASUREMENT_LOOPS; i++){
431 SINGLE_MEAS single_meas = single_measurement(Phase);
432 distance[i] = single_meas.distance;
433 angle[i] = single_meas.angle;
434 frequency[i] = single_meas.frequency;
435 current[i] = single_meas.current;
436
437 // Check if there is an error in the single measurement
438 if (single_meas.error != CALC_ERROR_NONE){
439 accu_meas.error_single = single_meas.error;
440 }
441 }
442 // Calculate the mean value of the distance
443 arm_mean_f32(distance, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.distance);
444 // Calculate standard deviation of the distance
445 arm_std_f32(distance, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.distance_std_dev);
446 // Calculate the mean value of the angle
447 arm_mean_f32(angle, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.angle);
448 // Calculate standard deviation of the angle
449 arm_std_f32(angle, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.angle_std_dev);
450 // Calculate the mean value of the frequency
451 arm_mean_f32(frequency, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.frequency);
452 // Calculate standard deviation of the frequency
453 arm_std_f32(frequency, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.frequency_std_dev);
454 // Calculate the mean value of the current
455 arm_mean_f32(current, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.current);
456 // Calculate standard deviation of the current
457 arm_std_f32(current, ACCURATE_MEASUREMENT_LOOPS, &accu_meas.current_std_dev);
458
459 // Check if the standard deviation is larger than 100
460 if (accu_meas.distance_std_dev > 100 || accu_meas.angle_std_dev > 100 || \
461 accu_meas.frequency_std_dev > 100 || accu_meas.current_std_dev > 100)
462 accu_meas.error_accu = CALC_ERROR_DEVIATION_TOO_HIGH;
463 else
464 accu_meas.error_accu = CALC_ERROR_NONE;
465
466 // Return the calculated values
467 return accu_meas;
468}

481DISTANCE_ANGLE calculate_distance_and_angle(float32_t signal_strength_r, float32_t signal_strength_l)
482{
483
484 // Create an instance of the DISTANCE_ANGLE structure
485 DISTANCE_ANGLE dist_angle;
486
487 // Calculate the distance using the coefficients and the actual signal strength
488 float32_t calc_distance = a_r + (b_r / signal_strength_r) + (c_r / POW2(signal_strength_r));
489 // Set the calculated values
490 dist_angle.distance_r = calc_distance;
491
492 // Calculate the distance using the coefficients and the actual signal strength
493 calc_distance = a_l + (b_l / signal_strength_l) + (c_l / POW2(signal_strength_l));
494 // Set the calculated values
495 dist_angle.distance_l = calc_distance;
496
497 // Calculate the distance
498 dist_angle.distance = (dist_angle.distance_l + dist_angle.distance_r) / 2;
499
500 // Calculate the angle using the trigonometry
501 dist_angle.angle = atanf((dist_angle.distance_l - dist_angle.distance_r) / (2 * D_P)) * (180 / PI);
502
503 // Return the calculated values
504 return dist_angle;
505}

558void calculate_coefficients_single_pad(float32_t s[], float32_t d[], float32_t* a, float32_t* b, float32_t* c)
559{
560 // s = x_0, x_1, x_2 d= y_0, y_1, y_2
561 // Calculate the coefficients for the distance approximation from an approximation based on a second degree polynomial
562 // a=((x_0^(2)*(x_1-x_2)*y_0-x_0*(x_1^(2)*y_1-x_2^(2)*y_2)+x_1*(x_1*y_1-x_2*y_2)*x_2)/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
563 // -> Result of Nspire CAS
564
565 *a = (POW2(s[0]) * (s[1] - s[2]) * d[0] - s[0] * (POW2(s[1]) * d[1] - POW2(s[2]) * d[2]) + s[1] * (s[1] * d[1] - s[2] * d[2]) * s[2])
566 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
567 // b = (((x_0^(2)*(x_1^(2)*(y_0-y_1)-x_2^(2)*(y_0-y_2))+x_1^(2)*x_2^(2)*(y_1-y_2)))/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
568 // -> Result of Nspire CAS
569
570 *b = (-(POW2(s[0]) * (POW2(s[1]) * (d[0] - d[1]) - POW2(s[2]) * (d[0] - d[2])) + POW2(s[1]) * POW2(s[2]) * (d[1] - d[2])))
571 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
572
573 // c=((x_0*(x_0*(x_1*(y_0-y_1)-x_2*(y_0-y_2))+x_1*x_2*(y_1-y_2))*x_1*x_2)/((x_0^(2)-x_0*(x_1+x_2)+x_1*x_2)*(x_1-x_2)))
574 // -> Result of Nspire CAS
575 *c = ((s[0] * (s[0] * (s[1] * (d[0] - d[1]) - s[2] * (d[0] - d[2])) + s[1] * s[2] * (d[1] - d[2])) * s[1] * s[2]))
576 / ((POW2(s[0]) - s[0] * (s[1] + s[2]) + s[1] * s[2]) * (s[1] - s[2]));
577}
